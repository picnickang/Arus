Add to SqliteFrameWriter:

public async Task RetentionLoopAsync(TimeSpan keepFor, CancellationToken ct)
{
    while (!ct.IsCancellationRequested)
    {
        await DeleteOlderThanAsync(DateTimeOffset.UtcNow - keepFor, ct);
        await Task.Delay(TimeSpan.FromHours(6), ct);
    }
}

private async Task DeleteOlderThanAsync(DateTimeOffset cutoff, CancellationToken ct)
{
    using var conn = new SqliteConnection($"Data Source={_dbPath}");
    await conn.OpenAsync(ct);

    using var cmd = conn.CreateCommand();
    cmd.CommandText = "PRAGMA busy_timeout=5000; DELETE FROM raw_frames WHERE unix_time_ms < $cutoff;";
    cmd.Parameters.AddWithValue("$cutoff", cutoff.ToUnixTimeMilliseconds());
    await cmd.ExecuteNonQueryAsync(ct);

    using var vacuum = conn.CreateCommand();
    vacuum.CommandText = "PRAGMA incremental_vacuum;";
    await vacuum.ExecuteNonQueryAsync(ct);
}

5) Build the SQLite → Postgres bridge (the “offline buffer” link)

You’re spot on: if your ARUS app uses Postgres in cloud mode, the pattern becomes:

Agent → local SQLite raw_frames → Node Bridge → decode/features → write to Postgres
…and optionally also write decoded into SQLite for local analytics.

Bridge responsibilities

Deliverables
	•	A Node service/module: server/services/sqlite-bridge/*
	•	Runs continuously (or cron-like interval)
	•	Safe idempotency + resumability
	•	Works offline: if Postgres unreachable, it keeps state and retries

Schema addition (recommended)

Add a lightweight cursor table in SQLite:

CREATE TABLE IF NOT EXISTS ingest_cursor (
  key TEXT PRIMARY KEY,
  last_raw_frame_id INTEGER NOT NULL
);
INSERT OR IGNORE INTO ingest_cursor(key, last_raw_frame_id) VALUES ('raw_frames', 0);

This avoids “marking” every raw frame row (which can add write amplification).

Bridge algorithm (robust + simple)
	1.	Read last_raw_frame_id
	2.	Fetch next batch: SELECT * FROM raw_frames WHERE id > ? ORDER BY id LIMIT ?
	3.	Decode (your existing decoder) → build events/signals/features
	4.	Write to Postgres inside a transaction with idempotency keys
	5.	On success:
	•	advance cursor to max id processed
	6.	If Postgres down:
	•	do nothing, retry later

Node bridge code skeleton (drop-in pattern)

import Database from "better-sqlite3";
import { pool } from "../db/postgres"; // your existing PG pool
import { decodeFrame } from "../telemetry/decode"; // your existing decoder

const BATCH_SIZE = 2000;

export async function runSqliteBridgeLoop(sqlitePath: string) {
  const db = new Database(sqlitePath);
  db.pragma("journal_mode = WAL");
  db.pragma("busy_timeout = 5000");

  db.exec(`
    CREATE TABLE IF NOT EXISTS ingest_cursor (
      key TEXT PRIMARY KEY,
      last_raw_frame_id INTEGER NOT NULL
    );
    INSERT OR IGNORE INTO ingest_cursor(key, last_raw_frame_id) VALUES ('raw_frames', 0);
  `);

  const getCursor = db.prepare(
    `SELECT last_raw_frame_id as lastId FROM ingest_cursor WHERE key='raw_frames'`
  );
  const setCursor = db.prepare(
    `UPDATE ingest_cursor SET last_raw_frame_id=? WHERE key='raw_frames'`
  );

  const fetchBatch = db.prepare(
    `SELECT id, unix_time_ms, source, protocol, payload, quality_flags
     FROM raw_frames
     WHERE id > ?
     ORDER BY id
     LIMIT ?`
  );

  while (true) {
    const { lastId } = getCursor.get() as { lastId: number };
    const rows = fetchBatch.all(lastId, BATCH_SIZE) as any[];

    if (rows.length === 0) {
      await sleep(500);
      continue;
    }

    // Decode
    const decoded = rows.flatMap(r => decodeFrame({
      id: r.id,
      ts: Number(r.unix_time_ms),
      source: r.source,
      protocol: r.protocol,
      payload: Buffer.from(r.payload),
      qualityFlags: Number(r.quality_flags),
    }));

    // Push to Postgres (idempotent)
    try {
      const client = await pool.connect();
      try {
        await client.query("BEGIN");

        // Example: write frames/signals with ON CONFLICT DO NOTHING
        // Use a deterministic idempotency key like `${source}:${protocol}:${id}`
        for (const item of decoded) {
          await client.query(
            `INSERT INTO telemetry_signals (idempotency_key, ts, signal_id, value, source)
             VALUES ($1,$2,$3,$4,$5)
             ON CONFLICT (idempotency_key) DO NOTHING`,
            [item.key, new Date(item.ts), item.signalId, item.value, item.source]
          );
        }

        await client.query("COMMIT");

        const maxId = rows[rows.length - 1].id;
        setCursor.run(maxId);
      } catch (e) {
        await client.query("ROLLBACK");
        throw e;
      } finally {
        client.release();
      }
    } catch (e) {
      // Postgres offline/unreachable: don’t advance cursor
      // Log + retry later
      await sleep(2000);
    }
  }
}

function sleep(ms: number) {
  return new Promise(res => setTimeout(res, ms));
}

Key production points
	•	Use ON CONFLICT DO NOTHING (or upsert) with an idempotency_key
	•	Cursor only advances after PG commit
	•	You can add a “local processed delete” step later:
	•	DELETE FROM raw_frames WHERE id <= lastId AND unix_time_ms < ... (optional)
	•	I’d keep raw for X days and let retention handle it
