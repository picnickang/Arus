syntax = "proto3";
package arus.telemetry;

option csharp_namespace = "Arus.Telemetry.Proto";

message Empty {}

message AgentStatus {
  string serviceVersion = 1;
  string status = 2;            // "running", "degraded"
  int64 unixTimeMs = 3;
  int64 framesWritten = 4;
  int64 lastFrameUnixTimeMs = 5;
}

message Ack { bool ok = 1; string message = 2; }

message SourceId { string id = 1; }

message SourceConfig {
  string id = 1;                 // "can0", "serial1"
  string type = 2;               // "CAN_J1939", "SERIAL", etc.
  string connectionString = 3;   // e.g. "COM3:115200" or adapter info
}

message IngestStats {
  int64 unixTimeMs = 1;
  int64 framesWritten = 2;
  int64 queueDepth = 3;
  int64 lastFrameUnixTimeMs = 4;
}

service TelemetryAgent {
  rpc GetStatus(Empty) returns (AgentStatus);
  rpc StartSource(SourceConfig) returns (Ack);
  rpc StopSource(SourceId) returns (Ack);
  rpc SubscribeStats(Empty) returns (stream IngestStats);
}

Storage/SqliteFrameWriter.cs (WAL + append-only raw_frames)

using Microsoft.Data.Sqlite;
using System.Collections.Concurrent;

namespace Arus.TelemetryAgent.Storage;

public class SqliteFrameWriter
{
    private readonly string _dbPath;
    private readonly ConcurrentQueue<RawFrame> _queue = new();
    private long _framesWritten = 0;
    private long _lastFrameUnixMs = 0;

    public SqliteFrameWriter(AgentPaths paths)
    {
        _dbPath = paths.DbPath;
        Directory.CreateDirectory(Path.GetDirectoryName(_dbPath)!);
        EnsureSchema();
    }

    public long FramesWritten => Interlocked.Read(ref _framesWritten);
    public long LastFrameUnixMs => Interlocked.Read(ref _lastFrameUnixMs);
    public int QueueDepth => _queue.Count;

    public void Enqueue(RawFrame f) => _queue.Enqueue(f);

    public async Task FlushLoopAsync(CancellationToken ct)
    {
        while (!ct.IsCancellationRequested)
        {
            await FlushBatchAsync(max: 500, ct);
            await Task.Delay(50, ct);
        }
    }

    private void EnsureSchema()
    {
        using var conn = new SqliteConnection($"Data Source={_dbPath}");
        conn.Open();

        using var cmd = conn.CreateCommand();
        cmd.CommandText = @"
PRAGMA journal_mode=WAL;
PRAGMA synchronous=NORMAL;

CREATE TABLE IF NOT EXISTS raw_frames (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  unix_time_ms INTEGER NOT NULL,
  source TEXT NOT NULL,
  protocol TEXT NOT NULL,
  payload BLOB NOT NULL,
  quality_flags INTEGER NOT NULL DEFAULT 0
);

CREATE INDEX IF NOT EXISTS idx_raw_frames_time ON raw_frames(unix_time_ms);
";
        cmd.ExecuteNonQuery();
    }

    private async Task FlushBatchAsync(int max, CancellationToken ct)
    {
        if (_queue.IsEmpty) return;

        var batch = new List<RawFrame>(max);
        while (batch.Count < max && _queue.TryDequeue(out var f)) batch.Add(f);
        if (batch.Count == 0) return;

        using var conn = new SqliteConnection($"Data Source={_dbPath}");
        await conn.OpenAsync(ct);

        using var tx = conn.BeginTransaction();
        var cmd = conn.CreateCommand();
        cmd.Transaction = tx;
        cmd.CommandText = @"
INSERT INTO raw_frames(unix_time_ms, source, protocol, payload, quality_flags)
VALUES ($t, $s, $p, $b, $q);";

        var pT = cmd.CreateParameter(); pT.ParameterName = "$t"; cmd.Parameters.Add(pT);
        var pS = cmd.CreateParameter(); pS.ParameterName = "$s"; cmd.Parameters.Add(pS);
        var pP = cmd.CreateParameter(); pP.ParameterName = "$p"; cmd.Parameters.Add(pP);
        var pB = cmd.CreateParameter(); pB.ParameterName = "$b"; cmd.Parameters.Add(pB);
        var pQ = cmd.CreateParameter(); pQ.ParameterName = "$q"; cmd.Parameters.Add(pQ);

        foreach (var f in batch)
        {
            pT.Value = f.UnixTimeMs;
            pS.Value = f.Source;
            pP.Value = f.Protocol;
            pB.Value = f.Payload;
            pQ.Value = f.QualityFlags;
            await cmd.ExecuteNonQueryAsync(ct);

            Interlocked.Exchange(ref _lastFrameUnixMs, f.UnixTimeMs);
        }

        tx.Commit();
        Interlocked.Add(ref _framesWritten, batch.Count);
    }
}

public record RawFrame(long UnixTimeMs, string Source, string Protocol, byte[] Payload, int QualityFlags = 0);

Worker.cs (heartbeat + stub “ingest loop”)

using System.Text.Json;
using Arus.TelemetryAgent.Storage;

namespace Arus.TelemetryAgent;

public class Worker : BackgroundService
{
    private readonly SqliteFrameWriter _writer;
    private readonly AgentPaths _paths;

    public Worker(SqliteFrameWriter writer, AgentPaths paths)
    {
        _writer = writer;
        _paths = paths;
        Directory.CreateDirectory(Path.GetDirectoryName(_paths.HeartbeatPath)!);
    }

    protected override async Task ExecuteAsync(CancellationToken stoppingToken)
    {
        // Start writer flush loop
        _ = _writer.FlushLoopAsync(stoppingToken);

        // Stub ingest loop (replace with real CAN/serial plugins)
        _ = Task.Run(async () =>
        {
            var rand = new Random();
            while (!stoppingToken.IsCancellationRequested)
            {
                var now = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds();
                var payload = new byte[16];
                rand.NextBytes(payload);

                _writer.Enqueue(new RawFrame(now, "sim0", "SIM", payload, 0));
                await Task.Delay(100, stoppingToken);
            }
        }, stoppingToken);

        // Heartbeat loop
        while (!stoppingToken.IsCancellationRequested)
        {
            var hb = new
            {
                unixTimeMs = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds(),
                framesWritten = _writer.FramesWritten,
                lastFrameUnixTimeMs = _writer.LastFrameUnixMs,
                queueDepth = _writer.QueueDepth,
                serviceVersion = "0.1.0"
            };

            var tmp = _paths.HeartbeatPath + ".tmp";
            await File.WriteAllTextAsync(tmp, JsonSerializer.Serialize(hb), stoppingToken);
            File.Move(tmp, _paths.HeartbeatPath, true);

            await Task.Delay(1000, stoppingToken);
        }
    }
}

import * as grpc from "@grpc/grpc-js";
import * as protoLoader from "@grpc/proto-loader";
import path from "path";

const PROTO_PATH = path.join(__dirname, "../protos/telemetry_agent.proto");

const pkgDef = protoLoader.loadSync(PROTO_PATH, {
  keepCase: true,
  longs: String,
  enums: String,
  defaults: true,
  oneofs: true,
});
const proto: any = grpc.loadPackageDefinition(pkgDef);

export function createAgentClient(address = "127.0.0.1:50051") {
  const client = new proto.arus.telemetry.TelemetryAgent(
    address,
    grpc.credentials.createInsecure()
  );
  return client;
}

export async function getAgentStatus(client: any) {
  return new Promise((resolve, reject) => {
    client.GetStatus({}, (err: any, resp: any) => {
      if (err) return reject(err);
      resolve(resp);
    });
  });
}

export function subscribeStats(client: any, onData: (s: any) => void, onError?: (e:any)=>void) {
  const stream = client.SubscribeStats({});
  stream.on("data", onData);
  stream.on("error", (e:any) => onError?.(e));
  stream.on("end", () => {});
  return stream;
}

3) Heartbeat fallback on Node side (if gRPC fails)

agentHeartbeat.ts

import fs from "fs";
import path from "path";

export function readHeartbeat(heartbeatPath: string) {
  try {
    const raw = fs.readFileSync(heartbeatPath, "utf-8");
    return JSON.parse(raw);
  } catch {
    return null;
  }
}

export function isAgentAlive(heartbeat: any, maxAgeMs = 5000) {
  if (!heartbeat?.unixTimeMs) return false;
  return Date.now() - Number(heartbeat.unixTimeMs) <= maxAgeMs;
}
