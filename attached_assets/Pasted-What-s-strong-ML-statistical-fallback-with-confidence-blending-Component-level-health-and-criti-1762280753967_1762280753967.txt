export type OpMode = 'DP'|'TRANSIT'|'HARBOR'|'STANDBY'|'UNKNOWN';

export function deriveOpMode(tags?: string[]): OpMode {
  if (!tags || tags.length === 0) return 'UNKNOWN';
  if (tags.includes('DP')) return 'DP';
  if (tags.includes('TRANSIT')) return 'TRANSIT';
  if (tags.includes('HARBOR')) return 'HARBOR';
  if (tags.includes('STANDBY')) return 'STANDBY';
  return 'UNKNOWN';
}

export function dataQualityScore(sampleCount: number, spanDays: number, missingPct: number, stalenessMin: number) {
  // Simple 0..1 quality: more samples, wider span, lower missing, fresher data => higher
  const qty = Math.min(1, sampleCount / 500);            // cap at 500 points
  const span = Math.min(1, spanDays / 30);               // cap at 30 days
  const miss = 1 - Math.min(1, missingPct);              // less missing is better
  const fresh = Math.max(0, 1 - (stalenessMin / 1440));  // 0 after 24h stale
  return Math.max(0, Math.min(1, 0.35*qty + 0.25*span + 0.25*miss + 0.15*fresh));
}

// Mode-aware threshold adjusters (tune empirically)
export function modeThresholdMultiplier(mode: OpMode) {
  switch(mode) {
    case 'DP': return 0.85;       // stricter (fails earlier)
    case 'TRANSIT': return 1.0;   // baseline
    case 'HARBOR': return 1.1;    // slightly more lenient
    case 'STANDBY': return 1.2;   // lenient
    default: return 1.0;
  }
}

// Simple isotonic-like calibration clamp (use real calibration later)
export function calibrateFailureProb(p: number, baseRate: number) {
  // pulls toward baseRate mildly to avoid overconfident extremes
  const alpha = 0.2;
  return Math.max(0.01, Math.min(0.99, (1-alpha)*p + alpha*baseRate));
}

D) Risk bands that also respect quality (small tweak)



Inside determineRiskLevel, lower bands when quality is poor. Easiest: pass a quality arg and widen buffers:

private determineRiskLevel(
  remainingDays: number, 
  failureProbability: number,
  healthIndex: number,
  quality: number // 0..1
): 'low'|'medium'|'high'|'critical' {
  const BUFFER = 0.05 + (1-quality)*0.05; // up to +5% buffer if poor quality
  const daysTilt = Math.round((1-quality)*3); // shrink days tolerance up to 3

  if (failureProbability > (0.7 - BUFFER) || remainingDays < (7 + 2 - daysTilt) || healthIndex < (30 + 5)) return 'critical';
  if (failureProbability > (0.4 - BUFFER) || remainingDays < (21 + 2 - daysTilt) || healthIndex < (60 + 5)) return 'high';
  if (failureProbability > (0.2 - BUFFER) || remainingDays < (35 + 2 - daysTilt) || healthIndex < (80 + 5)) return 'medium';
  return 'low';
}
