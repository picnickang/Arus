// Top-level helpers in this file
function parseIsoTimestamp(ts?: string): Date {
  if (!ts) return new Date();
  const d = new Date(ts);
  return isNaN(d.getTime()) ? new Date() : d;
}

// Round down to window start for any Date and size (ms)
function floorToWindow(d: Date, windowMs: number): Date {
  return new Date(Math.floor(d.getTime() / windowMs) * windowMs);
}

Windowed aggregation

 by datapoint timestamps (correctness & perf)

Process all relevant windows present in the buffer by bucketing datapoints by their window start (using each window size).

Prevent duplicate inserts with a single query per (equipmentId, sensorType, windowSize, windowStart) using SELECT … LIMIT 1.

After inserting for a window that is fully closed (i.e., end < now), prune those datapoints from buffer.

Actually use lastProcessedTime to avoid reprocessing very recent windows.



Replace processStreamAggregations() and createTimeWindowAggregate() with:

private async processStreamAggregations(): Promise<void> {
  try {
    const now = new Date();

    for (const [bufferKey, dataPoints] of this.dataBuffer.entries()) {
      if (dataPoints.length === 0) continue;
      const [equipmentId, sensorType] = bufferKey.split(':');

      for (const windowConfig of this.windowConfigs) {
        // Bucket by windowStart
        const buckets = new Map<number, Array<{ value:number; timestamp:Date; quality?:number }>>();
        for (const dp of dataPoints) {
          const ws = floorToWindow(dp.timestamp, windowConfig.windowSizeMs).getTime();
          if (!buckets.has(ws)) buckets.set(ws, []);
          buckets.get(ws)!.push(dp);
        }

        // Sort window starts ascending
        const starts = Array.from(buckets.keys()).sort((a,b)=>a-b);
        for (const wsMillis of starts) {
          const windowStart = new Date(wsMillis);
          const windowEnd = new Date(wsMillis + windowConfig.windowSizeMs);
          // Only finalize windows that have closed
          if (windowEnd > now) continue;

          const bucket = buckets.get(wsMillis)!;
          await this.createOrSkipAggregate(equipmentId, sensorType, bucket, windowConfig, windowStart, windowEnd);

          // prune buffer datapoints for this finalized window
          const arr = this.dataBuffer.get(bufferKey)!;
          this.dataBuffer.set(
            bufferKey,
            arr.filter(dp => !(dp.timestamp >= windowStart && dp.timestamp < windowEnd))
          );
          this.lastProcessedTime.set(`${bufferKey}:${windowConfig.windowSize}`, windowEnd);
        }
      }
    }
  } catch (error) {
    console.error('[MQTT Ingestion] Error processing stream aggregations:', error);
  }
}

private async createOrSkipAggregate(
  equipmentId: string,
  sensorType: string,
  windowData: Array<{ value:number; timestamp:Date; quality?:number }>,
  windowConfig: WindowConfig,
  windowStart: Date,
  windowEnd: Date
): Promise<void> {
  if (windowData.length === 0) return;

  // Check if aggregate exists
  const existing = await db
    .select()
    .from(telemetryAggregates)
    .where(
      and(
        eq(telemetryAggregates.equipmentId, equipmentId),
        eq(telemetryAggregates.sensorType, sensorType),
        eq(telemetryAggregates.timeWindow, windowConfig.windowSize),
        eq(telemetryAggregates.windowStart, windowStart)
      )
    )
    .limit(1);

  if (existing.length > 0) return;

  const values = windowData.map(d => d.value);
  const avgValue = values.reduce((s,v)=>s+v,0) / values.length;
  const minValue = Math.min(...values);
  const maxValue = Math.max(...values);
  const variance = values.reduce((s,v)=>s+Math.pow(v-avgValue,2),0) / values.length;
  const stdDev = Math.sqrt(variance);
  const qualityScores = windowData.map(d => d.quality ?? 1.0);
  const avgQuality = qualityScores.reduce((s,q)=>s+q,0) / qualityScores.length;

  // Insert with retry for transient failures
  await retryWithBackoff(
    () => db.insert(telemetryAggregates).values({
      equipmentId,
      sensorType,
      timeWindow: windowConfig.windowSize,
      windowStart,
      windowEnd,
      avgValue,
      minValue,
      maxValue,
      stdDev,
      sampleCount: windowData.length,
      anomalyScore: null,
      qualityScore: avgQuality,
      metadata: {
        windowSizeMs: windowConfig.windowSizeMs,
        aggregationMethod: windowConfig.aggregationFunction,
        variance
      }
    }),
    {
      maxRetries: 3,
      initialDelayMs: 100,
      maxDelayMs: 2000,
      shouldRetry: (err) => isRetryableError?.(err) ?? true
    }
  );

  this.emit('aggregate_created', {
    equipmentId, sensorType,
    timeWindow: windowConfig.windowSize,
    windowStart, avgValue,
    sampleCount: windowData.length,
    qualityScore: avgQuality
  });
}
