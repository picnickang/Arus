import { db } from "../db";
import { sql } from "drizzle-orm";

export async function nextSeq(vesselId: string, entity: string): Promise<number> {
  const result = await db.execute(sql`
    INSERT INTO entity_offsets (vessel_id, entity, seq)
    VALUES (${vesselId}, ${entity}, 1)
    ON CONFLICT (vessel_id, entity)
    DO UPDATE SET seq = entity_offsets.seq + 1
    RETURNING seq;
  `);
  const row = Array.isArray((result as any)?.rows) ? (result as any).rows[0] : (result as any)[0];
  return Number(row.seq);
}

c) Add a raw publish helper

async publishRaw(
  topicStr: string,
  payload: unknown | string,
  opts: { qos?: 0|1|2; retain?: boolean } = {}
): Promise<void> {
  const qos = opts.qos ?? this.config.qosLevel;
  const retain = opts.retain ?? false;
  const body = typeof payload === "string" ? payload : JSON.stringify(payload);

  if (this.isConnected && this.client) {
    await new Promise<void>((resolve) => {
      this.client!.publish(topicStr, body, { qos, retain }, (err) => {
        if (err) {
          this.queueMessage({ topic: topicStr, payload, qos, retain });
          this.metrics.publishFailures++; incrementMqttPublishFailures();
        } else {
          this.metrics.messagesPublished++; incrementMqttMessagesPublished('raw', 'publish', qos);
        }
        resolve(); // never reject to avoid caller retry loops
      });
    });
  } else {
    this.queueMessage({ topic: topicStr, payload, qos, retain });
    this.emit('message_queued', { topic: topicStr, qos });
  }
}

private async flushMessageQueue() {
  if (!this.client || !this.isConnected) return;

  const now = Date.now();
  const ready: Queued[] = [];
  const later: Queued[] = [];

  while (this.messageQueue.length) {
    const m = this.messageQueue.shift()!;
    if (!m.nextAt || m.nextAt <= now) ready.push(m);
    else later.push(m);
  }
  this.messageQueue.unshift(...later);
  if (ready.length === 0) return;

  console.log(`[MQTT Reliable Sync] Flushing ${ready.length} queued messages`);
  let success = 0, failed = 0;

  for (const m of ready) {
    try {
      await new Promise<void>((resolve, reject) => {
        this.client!.publish(
          m.topic,
          typeof m.payload === "string" ? m.payload : JSON.stringify(m.payload),
          { qos: m.qos, retain: m.retain },
          (err) => (err ? reject(err) : resolve())
        );
      });
      success++;
    } catch {
      failed++;
      const attempts = (m.attempts ?? 0) + 1;
      const delay = Math.min(60_000, 2 ** Math.min(attempts, 10) * 100);
      m.attempts = attempts;
      m.nextAt = Date.now() + delay;
      this.messageQueue.push(m);
    }
  }

  if (success > 0) {
    this.metrics.queueFlushes++; incrementMqttQueueFlushes();
    this.emit('queue_flushed', { sent: success, failed });
  }

  updateMqttMetrics({
    currentQueueSize: this.messageQueue.length,
    queueUtilization: (this.messageQueue.length / this.config.maxQueueSize) * 100
  });
}

private seen = new Map<string, number>();
private maxSeen = 5000;

private isDuplicate(id?: string) {
  if (!id) return false;
  if (this.seen.has(id)) return true;
  this.seen.set(id, Date.now());
  if (this.seen.size > this.maxSeen) {
    const cutoff = Date.now() - 60_000;
    for (const [k, v] of this.seen) if (v < cutoff) this.seen.delete(k);
  }
  return false;
}
