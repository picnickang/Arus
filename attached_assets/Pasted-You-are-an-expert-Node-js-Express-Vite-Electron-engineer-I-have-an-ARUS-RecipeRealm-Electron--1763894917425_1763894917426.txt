import path from 'path';
import express from 'express';
import fs from 'fs';

const app = express();

// 1) API routes first (adjust if you have a different router)
app.use('/api', apiRouter);

// 2) Decide staticRoot based on existing logic (you already do candidate roots: dist, dist/public, client/dist)
const staticRoot = /* the one you already resolve, e.g. path.resolve(__dirname, '../dist') */;

// 3) Static files
app.use(
  express.static(staticRoot, {
    index: 'index.html',
    setHeaders: (res, filePath) => {
      // Ensure correct MIME types (helps avoid strict MIME type issues)
      if (filePath.endsWith('.css')) {
        res.type('text/css');
      }
      if (filePath.endsWith('.js')) {
        res.type('application/javascript');
      }
    },
  }),
);

// 4) SPA fallback: ONLY for non-API routes that do NOT look like files
app.get('*', (req, res, next) => {
  // Let /api/* and any URL with an extension (.css, .js, .png, etc.) skip the SPA fallback
  if (req.path.startsWith('/api/') || path.extname(req.path)) {
    return next();
  }

  res.sendFile(path.join(staticRoot, 'index.html'));
});

// 5) Final 404 handler for anything that truly doesn't exist
app.use((req, res) => {
  res.status(404).json({
    error: 'Not Found',
    path: req.path,
  });
});
Important details:

express.static(staticRoot) must be before app.get('*', ...).

The SPA fallback must not match /assets/... or any URL that has a file extension.
