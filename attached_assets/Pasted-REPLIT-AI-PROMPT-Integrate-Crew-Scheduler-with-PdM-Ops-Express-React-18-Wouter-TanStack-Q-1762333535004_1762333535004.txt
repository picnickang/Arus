type EventName = "pdm.anomaly.created" | "pdm.maintenance.window" | "pdm.rul.updated";
type Handler = (payload: any) => Promise<void> | void;

const handlers = new Map<EventName, Handler[]>();
export function on(name: EventName, h: Handler){ handlers.set(name, [...(handlers.get(name)||[]), h]); }
export async function emit(name: EventName, payload: any){ for(const h of (handlers.get(name)||[])) await h(payload); }
Wire PdM to emit:

After RUL calc: emit("pdm.rul.updated", { orgId, vesselId, remainingDays, riskLevel })

On anomaly creation: emit("pdm.anomaly.created", { orgId, vesselId, severity, window })

On maintenance window creation: emit("pdm.maintenance.window", { orgId, vesselId, start, end })



Create auto-replan policy:



server/scheduler/auto-replan.ts

import { on } from "../events/bus.js";
import { planAndMaybeExecute } from "./controller.js";

const RUL_DAYS_CRITICAL = Number(process.env.SCHED_RUL_DAYS_CRITICAL ?? 9);
const RISK_REPLAN_LEVEL = (process.env.SCHED_RISK_REPLAN_LEVEL ?? "high").toLowerCase();

function riskToRank(r: string){ return ({low:0,medium:1,high:2,critical:3} as any)[r] ?? 0; }

on("pdm.rul.updated", async ({ orgId, vesselId, remainingDays, riskLevel }) => {
  if (remainingDays <= RUL_DAYS_CRITICAL || riskToRank(riskLevel) >= riskToRank(RISK_REPLAN_LEVEL)) {
    await planAndMaybeExecute({ orgId, days: 7, vessels: [vesselId], mode: "auto" });
  }
});

on("pdm.anomaly.created", async ({ orgId, vesselId, severity }) => {
  if (severity === "high" || severity === "critical") {
    await planAndMaybeExecute({ orgId, days: 7, vessels: [vesselId], mode: "auto" });
  }
});

on("pdm.maintenance.window", async ({ orgId, vesselId }) => {
  await planAndMaybeExecute({ orgId, days: 7, vessels: [vesselId], mode: "auto" });
});

server/scheduler/controller.ts

import crypto from "node:crypto";
import { db } from "../db/connection.js";
import { schedulerRuns, scheduleAssignments, scheduleUnfilled } from "../db/schema.scheduler.js";
import { planShifts, generateDays } from "../scheduler/crew-scheduler.js"; // your file
import { eq } from "drizzle-orm";
import { rulRemainingDays } from "../observability/rul-metrics.js"; // optional

type PlanParams = { orgId: string; from?: string; days?: number; vessels?: string[]; mode?: "dry_run"|"execute"|"auto" };

export async function planAndMaybeExecute({ orgId, from, days=7, vessels, mode="dry_run" }: PlanParams){
  const since = from ?? new Date().toISOString().slice(0,10);
  // Load inputs: shifts, crew, leaves, existing
  const daysArr = generateDays(since, days);

  // TODO: replace with real repos for your domain
  const shifts = await loadShiftTemplates(orgId, vessels);      // includes skillRequired, needed, vesselId, start, end, role
  const crew = await loadCrewWithSkills(orgId);                 // includes id, skills, vesselId, rank, minRestH, maxHours7d, active, preferences?
  const leaves = await loadCrewLeaves(orgId);
  const existing = await loadExistingAssignments(orgId, since, daysArr.at(-1)!);

  const inputHash = crypto.createHash("sha256")
    .update(JSON.stringify({ daysArr, shifts, crew, leaves, existing })).digest("hex");

  const run = await db.insert(schedulerRuns).values({
    orgId, startedAt: new Date(), mode, inputHash
  }).returning({ id: schedulerRuns.id });

  const t0 = Date.now();
  const { scheduled, unfilled } = planShifts(daysArr, shifts, crew, leaves, existing);
  const ms = Date.now()-t0;

  // persist results
  if (mode !== "dry_run") {
    await db.insert(scheduleAssignments).values(scheduled.map(a => ({
      runId: run[0].id, orgId, ...a, executed: mode !== "dry_run"
    })));
  }
  await db.insert(scheduleUnfilled).values(unfilled.map(u => ({ runId: run[0].id, orgId, ...u })));

  const stats = {
    duration_ms: ms,
    assigned: scheduled.length,
    unfilled: unfilled.reduce((s,u)=>s+u.need,0),
    reasons: aggregateReasons(unfilled.map(u => u.reason))
  };

  await db.update(schedulerRuns).set({
    finishedAt: new Date(), success: true, stats
  }).where(eq(schedulerRuns.id, run[0].id));

  return { runId: run[0].id, mode, stats, scheduled, unfilled };
}

function aggregateReasons(strings: string[]){
  const m = new Map<string, number>();
  for (const s of strings){ m.set(s, (m.get(s)||0)+1); }
  return Array.from(m, ([k,v])=>({reason:k,count:v}));

Routes (Express):



server/routes/schedule.routes.ts

import { Router } from "express";
import { planAndMaybeExecute } from "../scheduler/controller.js";
import { db } from "../db/connection.js";
import { schedulerRuns, scheduleAssignments, scheduleUnfilled } from "../db/schema.scheduler.js";
import { and, eq, gte, lte } from "drizzle-orm";

export const scheduleRouter = Router();

scheduleRouter.post("/plan", async (req, res) => {
  const { orgId, from, days, vessels, mode } = req.body ?? {};
  if (!orgId) return res.status(400).json({ error: "orgId required" });
  const out = await planAndMaybeExecute({ orgId, from, days, vessels, mode });
  res.json(out);
});

scheduleRouter.get("/runs", async (req, res) => {
  const { orgId } = req.query as any;
  if (!orgId) return res.status(400).json({ error: "orgId required" });
  const rows = await db.select().from(schedulerRuns).where(eq(schedulerRuns.orgId, orgId)).orderBy(schedulerRuns.startedAt);
  res.json(rows);
});

scheduleRouter.get("/assignments", async (req, res) => {
  const { orgId, from, to } = req.query as any;
  if (!orgId || !from || !to) return res.status(400).json({ error: "orgId, from, to required" });
  const rows = await db.select().from(scheduleAssignments).where(
    and(eq(scheduleAssignments.orgId, orgId), gte(scheduleAssignments.start, new Date(from)), lte(scheduleAssignments.end, new Date(to)))
  );
  res.json(rows);
});

scheduleRouter.get("/unfilled", async (req, res) => {
  const { orgId, runId } = req.query as any;
  if (!orgId) return res.status(400).json({ error: "orgId required" });
  const where = runId ? and(eq(scheduleUnfilled.orgId, orgId), eq(scheduleUnfilled.runId, runId))
                      : eq(scheduleUnfilled.orgId, orgId);
  const rows = await db.select().from(scheduleUnfilled).where(where);
  res.json(rows);
});

ScheduleBoard (table + run controls):



client/views/ScheduleBoard.tsx

import { useMutation, useQuery } from "@tanstack/react-query";
import { useState } from "react";

export default function ScheduleBoard(){
  const [orgId] = useState("ORG1");
  const [days, setDays] = useState(7);
  const from = new Date().toISOString().slice(0,10);

  const runs = useQuery({
    queryKey:["sched-runs", orgId],
    queryFn: async()=> (await fetch(`/api/schedule/runs?orgId=${orgId}`)).json()
  });

  const assignments = useQuery({
    queryKey:["sched-assignments", orgId, from, days],
    queryFn: async()=>{
      const to = new Date(Date.now()+days*864e5).toISOString();
      return (await fetch(`/api/schedule/assignments?orgId=${orgId}&from=${from}&to=${to}`)).json();
    }
  });

  const unfilled = useQuery({
    queryKey:["sched-unfilled", orgId],
    queryFn: async()=> (await fetch(`/api/schedule/unfilled?orgId=${orgId}`)).json()
  });

  const plan = useMutation({
    mutationFn: async (mode:"dry_run"|"execute")=>{
      return await fetch(`/api/schedule/plan`,{ method:"POST", headers:{ "content-type":"application/json" },
        body: JSON.stringify({ orgId, from, days, mode })}).then(r=>r.json());
    },
    onSuccess: ()=>{ runs.refetch(); assignments.refetch(); unfilled.refetch(); }
  });

  return (
    <div className="p-4 space-y-4">
      <div className="flex gap-2 items-center">
        <button onClick={()=>plan.mutate("dry_run")} className="px-3 py-2 rounded bg-slate-800">Dry-run</button>
        <button onClick={()=>plan.mutate("execute")} className="px-3 py-2 rounded bg-indigo-600">Execute</button>
        <span>Days: </span>
        <input type="number" min={1} max={28} value={days} onChange={e=>setDays(Number(e.target.value))} className="w-20 bg-slate-900 p-2 rounded"/>
      </div>

      <div className="grid grid-cols-2 gap-4">
        <div>
          <h3 className="font-semibold mb-2">Assignments</h3>
          <div className="max-h-[360px] overflow-auto text-sm border rounded">
            {(assignments.data||[]).map((a:any)=>(
              <div key={a.id} className="px-2 py-1 border-b">
                {a.date} • {a.shiftId} • {a.crewId} • {a.vesselId ?? "-"} • {new Date(a.start).toLocaleString()} → {new Date(a.end).toLocaleString()}
              </div>
            ))}
          </div>
        </div>
        <div>
          <h3 className="font-semibold mb-2">Unfilled (reasons)</h3>
          <div className="max-h-[360px] overflow-auto text-sm border rounded">
            {(unfilled.data||[]).map((u:any)=>(
              <div key={u.id} className="px-2 py-1 border-b">
                {u.day} • {u.shiftId} → need {u.need} | {u.reason}
              </div>
            ))}
          </div>
        </div>
      </div>

      <div>
        <h3 className="font-semibold mb-2">Runs</h3>
        <div className="text-sm border rounded max-h-[240px] overflow-auto">
          {(runs.data||[]).map((r:any)=>(
            <div key={r.id} className="px-2 py-1 border-b">
              {r.mode} • {r.success ? "OK":"FAIL"} • {new Date(r.startedAt).toLocaleString()} → {r.finishedAt ? new Date(r.finishedAt).toLocaleString() : "…"}
              {r.stats ? <span className="ml-2 opacity-80">assigned {r.stats.assigned}, unfilled {r.stats.unfilled}</span> : null}
            </div>
          ))}
        </div>
      </div>
    </div>
  );
}

import { useParams } from "wouter";
import { useQuery } from "@tanstack/react-query";

    queryKey:["rul-series", orgId, vesselId],

  const availability = useQuery({
    queryKey:["crew-availability", orgId, vesselId, from, to],

  return (
    <div className="p-4 space-y-4">
      <h2 className="text-xl font-semibold">Vessel {vesselId} — RUL & Crew Availability</h2>
      <div className="text-sm opacity-80">RUL (days) with crewed windows overlay</div>
      <div className="border rounded p-3">
        {/* Replace with your charting component; below is a simple readout */}

          <div>
            <h4 className="font-semibold mb-1">RUL Points</h4>
