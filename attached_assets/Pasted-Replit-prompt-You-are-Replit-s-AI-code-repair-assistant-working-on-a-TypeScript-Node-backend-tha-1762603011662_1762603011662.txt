import { db } from "./db";
import { sensorTemplates, organizations } from "@shared/schema";
import { eq, and, isNull } from "drizzle-orm";
import * as fs from "fs/promises";
import * as path from "path";

interface JsonTemplate {
  id: string;
  kind: string;
  unit: string;
  fields: Record<string, any>;
  notes?: string;
}

interface TemplatesFile {
  templates: JsonTemplate[];
}

// Map JSON template fields to database schema
function mapJsonTemplateToDbTemplate(jsonTemplate: JsonTemplate) {
  // Extract equipment type from fields if present
  const equipmentType = jsonTemplate.fields.equipment_type;
  const equipmentTypes = equipmentType ? [equipmentType] : null;
  
  // Generate human-readable name from template ID
  const name = jsonTemplate.id
    .replace(/-/g, ' ')
    .replace(/\b\w/g, l => l.toUpperCase());
  
  return {
    orgId: null, // System templates are global (not tied to any org)
    templateId: jsonTemplate.id,
    name,
    kind: jsonTemplate.kind,
    unit: jsonTemplate.unit,
    equipmentTypes,
    fields: jsonTemplate.fields,
    notes: jsonTemplate.notes || null,
    isSystemDefault: true,
    createdBy: 'system-seed'
  };
}

async function seedSensorTemplates() {
  console.log('🌱 Starting sensor template seeding...');
  console.log('📌 Creating global system templates (orgId = null)');
  
  try {
    // Read sensor templates from JSON file
    const jsonPath = path.join(process.cwd(), 'server', 'config', 'sensor_templates.json');
    const jsonContent = await fs.readFile(jsonPath, 'utf-8');
    const templatesData: TemplatesFile = JSON.parse(jsonContent);
    
    console.log(`📖 Found ${templatesData.templates.length} templates in JSON file`);
    
    // Upsert each template as a global system template (orgId = null)
    let created = 0;
    let updated = 0;
    
    for (const jsonTemplate of templatesData.templates) {
      const dbTemplate = mapJsonTemplateToDbTemplate(jsonTemplate);
      
      // Check if system template already exists (orgId IS NULL)
      const existing = await db.select()
        .from(sensorTemplates)
        .where(
          and(
            isNull(sensorTemplates.orgId),
            eq(sensorTemplates.templateId, dbTemplate.templateId)
          )
        )
        .limit(1);
      
      if (existing.length === 0) {
        // Insert new system template
        await db.insert(sensorTemplates).values(dbTemplate);
        created++;
        console.log(`  ✓ Created: ${dbTemplate.templateId} (${dbTemplate.kind})`);
      } else {
        // Update existing system template (preserve ID but update other fields)
        await db.update(sensorTemplates)
          .set({
            name: dbTemplate.name,
            kind: dbTemplate.kind,
            unit: dbTemplate.unit,
            equipmentTypes: dbTemplate.equipmentTypes,
            fields: dbTemplate.fields,
            notes: dbTemplate.notes,
            updatedAt: new Date()
          })
          .where(
            and(
              isNull(sensorTemplates.orgId),
              eq(sensorTemplates.templateId, dbTemplate.templateId)
            )
          );
        updated++;
        console.log(`  ↻ Updated: ${dbTemplate.templateId} (${dbTemplate.kind})`);
      }
    }
    
    console.log(`\n✅ Seeding complete!`);
    console.log(`   Created: ${created} templates`);
    console.log(`   Updated: ${updated} templates`);
    console.log(`   Total:   ${created + updated} templates`);
    
  } catch (error) {
    console.error('❌ Seeding failed:', error);
    throw error;
  }
}

// Run seeding if executed directly
if (import.meta.url === `file://${process.argv[1]}`) {
  seedSensorTemplates()

    })

    });
}
