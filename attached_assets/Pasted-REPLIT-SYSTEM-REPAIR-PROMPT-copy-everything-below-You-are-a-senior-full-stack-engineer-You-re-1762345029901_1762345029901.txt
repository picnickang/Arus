import { ZodSchema } from 'zod';
import type { Request, Response, NextFunction } from 'express';

export const validateBody = (schema: ZodSchema) => (req: Request, res: Response, next: NextFunction) => {
  const parsed = schema.safeParse(req.body);
  if (!parsed.success) return res.status(400).json({ code:'BAD_REQUEST', message:'Validation failed', details: parsed.error.flatten() });
  req.body = parsed.data;
  next();
};

Add: server/middleware/csrf.ts

export const csrfGuard = (req,res,next) => {
  if (['POST','PUT','PATCH','DELETE'].includes(req.method)) {
    const token = req.headers['x-csrf-token'];
    if (!token || token !== process.env.CSRF_TOKEN) return res.status(403).json({ code:'CSRF', message:'Invalid CSRF token' });
  }
  next();
};
Add: server/middleware/rateLimit.ts (simple in-memory per IP for admin)

const hits = new Map<string,{count:number, ts:number}>();
export const rateLimit = (limit=60, windowMs=60_000) => (req,res,next)=>{
  const key = `${req.ip}:admin`;
  const now = Date.now();
  const rec = hits.get(key) || {count:0, ts:now};
  if (now - rec.ts > windowMs) { rec.count=0; rec.ts=now; }
  rec.count++;
  hits.set(key, rec);
  if (rec.count > limit) return res.status(429).json({ code:'RATE_LIMIT', message:'Too many requests' });
  next();
};

// scripts/bench-seed.ts
// Synthetic scenarios for LP optimizer + cost engine
// Usage: npx tsx scripts/bench-seed.ts small|medium|stress --base http://localhost:3000 --org ORG_1

type Size = "small" | "medium" | "stress";

interface CrewAvail {
  crewMember: string;
  availableDays: string[];
  maxHoursPerDay: number;
  skillLevel: number;
  hourlyRate: number;
}

interface JobPart { partId: string; quantity: number; unitCost: number; }

interface Job {
  id: string;
  equipmentId: string;
  equipmentName: string;
  maintenanceType: "preventive" | "corrective";
  priority: 1 | 2 | 3 | 4;
  estimatedDuration: number; // minutes
  requiredSkillLevel: number;
  parts: JobPart[];
  preferredDate?: Date;
  deadline?: Date;
  dependencies?: string[];
}

interface Constraints {
  maxDailyWorkHours: number;
  maxConcurrentJobs: number;
  crewAvailability: CrewAvail[];
  partsBudget: number;
  timeHorizonDays: number;
  priorityWeights: { critical: number; high: number; medium: number; low: number; };
}

interface ScenarioPayload {
  name: string;
  orgId: string;
  vessels: { id: string; name: string; }[];
  equipment: { id: string; name: string; vesselId: string; type: string; }[];
  constraints: Constraints;
  jobs: Job[];
}

const WEEKDAYS = ["Monday","Tuesday","Wednesday","Thursday","Friday"];
const ALLDAYS = ["Sunday","Monday","Tuesday","Wednesday","Thursday","Friday","Saturday"];

const pick = <T>(arr: T[], n: number) => arr.slice(0, n);
const rand = (min: number, max: number) => Math.floor(Math.random()*(max-min+1))+min;
const choice = <T>(arr: T[]) => arr[rand(0, arr.length-1)];

function crewFactory(n: number, allDays: string[], baseRate=70): CrewAvail[] {
  return Array.from({ length: n }).map((_, i) => {
    const skill = (i % 5 === 0) ? 4 : (i % 3 === 0) ? 3 : 2;
    const rate = baseRate + skill*10 + rand(0,10);
    const avail = (i % 4 === 0) ? ALLDAYS : (i % 2 === 0) ? allDays : pick(allDays, 3);
    return {
      crewMember: `tech_${String(i+1).padStart(2,"0")}`,
      availableDays: avail,
      maxHoursPerDay: (i % 3 === 0) ? 10 : 8,
      skillLevel: skill,
      hourlyRate: rate
    };
  });
}

function vesselEquipmentFactory(vessels: number, eqPerVessel: number) {
  const vs = Array.from({ length: vessels }).map((_, i) => ({
    id: `V${i+1}`, name: `Vessel-${i+1}`
  }));
  const eq: { id: string; name: string; vesselId: string; type: string }[] = [];
  vs.forEach(v => {
    for (let i=1;i<=eqPerVessel;i++){
      const type = (i%3===0) ? "genset" : (i%2===0) ? "pump" : "engine";
      eq.push({ id:`E_${v.id}_${i}`, name:`${v.name}-${type}-${i}`, vesselId: v.id, type });
    }
  });
  return { vessels: vs, equipment: eq };
}

function partsFor(type: string, maint: "preventive"|"corrective"): JobPart[] {
  if (maint === "preventive") {
    if (type === "engine") return [{ partId:"filter-oil", quantity:1, unitCost:25 },{ partId:"oil", quantity:5, unitCost:8 }];
    if (type === "pump")   return [{ partId:"seal-kit", quantity:1, unitCost:40 }];
    return [{ partId:"inspect-kit", quantity:1, unitCost:20 }];
  } else {
    return [{ partId:"misc", quantity:1, unitCost: rand(80,180) }];
  }
}

function jobsFactory(
  count: number,
  equipment: { id:string; name:string; vesselId:string; type:string }[]
): Job[] {
  return Array.from({ length: count }).map((_, i) => {
    const eq = choice(equipment);
    const priority = (i % 10 === 0) ? 1 : (i % 5 === 0) ? 2 : (i % 3 === 0) ? 3 : 4;
    const maint: "preventive"|"corrective" = (i % 4 === 0) ? "corrective" : "preventive";
    const dur = (maint === "corrective") ? rand(120,300) : rand(60,180);
    const skill = (eq.type === "engine" && maint === "corrective") ? 4 : (eq.type === "engine" ? 3 : (maint==="corrective"?3:2));
    const start = new Date(); const dd = new Date(start.getTime()+rand(0,5)*86400000);
    const deadline = new Date(dd.getTime()+rand(2,7)*86400000);
    return {
      id: `JOB_${String(i+1).padStart(3,"0")}`,
      equipmentId: eq.id,
      equipmentName: eq.name,
      maintenanceType: maint,
      priority: priority as 1|2|3|4,
      estimatedDuration: dur,
      requiredSkillLevel: skill,
      parts: partsFor(eq.type, maint),
      preferredDate: dd,
      deadline
    };
  });
}

function scenario(size: Size, orgId: string): ScenarioPayload {
  if (size==="small") {
    const { vessels, equipment } = vesselEquipmentFactory(2, 3); // 2 vessels x 3 eq
    return {
      name: "bench_small",
      orgId,
      vessels, equipment,
      constraints: {
        maxDailyWorkHours: 10,
        maxConcurrentJobs: 2,
        crewAvailability: crewFactory(6, WEEKDAYS, 65),
        partsBudget: 3000,
        timeHorizonDays: 7,
        priorityWeights: { critical: 200, high: 60, medium: 25, low: 10 }
      },
      jobs: jobsFactory(10, equipment)
    };
  }
  if (size==="medium") {
    const { vessels, equipment } = vesselEquipmentFactory(3, 5);
    return {
      name: "bench_medium",
      orgId,
      vessels, equipment,
      constraints: {
        maxDailyWorkHours: 10,
        maxConcurrentJobs: 3,
        crewAvailability: crewFactory(12, WEEKDAYS, 70),
        partsBudget: 12000,
        timeHorizonDays: 14,
        priorityWeights: { critical: 220, high: 70, medium: 30, low: 12 }
      },
      jobs: jobsFactory(40, equipment)
    };
  }
  // stress
  const { vessels, equipment } = vesselEquipmentFactory(5, 6);
  return {
    name: "bench_stress",
    orgId,
    vessels, equipment,
    constraints: {
      maxDailyWorkHours: 10,
      maxConcurrentJobs: 5,
      crewAvailability: crewFactory(24, WEEKDAYS, 75),
      partsBudget: 60000,
      timeHorizonDays: 28,
      priorityWeights: { critical: 250, high: 90, medium: 40, low: 15 }
    },
    jobs: jobsFactory(150, equipment)
  };
}

async function main() {
  const sizeArg = (process.argv[2] as Size) || "small";
  const base = (process.argv.includes("--base") ? process.argv[process.argv.indexOf("--base")+1] : "http://localhost:3000").replace(/\/+$/,"");
  const org = process.argv.includes("--org") ? process.argv[process.argv.indexOf("--org")+1] : "ORG_DEMO";
  if (!["small","medium","stress"].includes(sizeArg)) {
    console.error("Usage: npx tsx scripts/bench-seed.ts small|medium|stress --base http://localhost:3000 --org ORG_1");
    process.exit(1);
  }
  const payload = scenario(sizeArg, org);

  // Optional: write to disk for inspection
  const fs = await import("node:fs");
  const outPath = `.bench/${payload.name}.json`;
  fs.mkdirSync(".bench",{recursive:true});
  fs.writeFileSync(outPath, JSON.stringify(payload, null, 2));
  console.log(`✓ Wrote ${outPath}`);

  // POST to your (expected) seed endpoint; fallback to optimizer if you prefer
  const res = await fetch(`${base}/api/experiments/seed`, {
    method: "POST",
    headers: { "content-type":"application/json" },
    body: JSON.stringify(payload)
  }).catch(() => null);

  if (res && res.ok) {
    const body = await res.json().catch(()=> ({}));
    console.log(`✓ Seeded '${payload.name}' → id:`, body?.id ?? "(ok)");
  } else {
    console.warn("⚠ Could not reach /api/experiments/seed. You can still use the file:");
    console.warn(`   ${outPath}`);
    console.warn("   Or POST it manually to your optimizer endpoint (/api/optimizer/lp/run) if preferred.");
  }
}

main().catch(e => { console.error(e); process.exit(1); });
2) Add a tiny seed API (if you don’t have one yet)

// routes/experiments.ts
import { Router } from "express";
import { db } from "../db";
import { optimizationExperiments } from "@shared/schema";

const r = Router();

r.post("/seed", async (req, res) => {
  try {
    const { name, orgId, vessels, equipment, constraints, jobs } = req.body ?? {};
    if (!name || !orgId || !constraints || !Array.isArray(jobs)) {
      return res.status(400).json({ error: "Invalid payload" });
    }
    // Persist a single row summarizing the experiment config
    const [row] = await db.insert(optimizationExperiments).values({
      orgId,
      name,
      configJson: JSON.stringify({ vessels, equipment, constraints, jobs }),
      createdAt: new Date(),
    }).returning({ id: optimizationExperiments.id });
    return res.json({ id: row.id, ok: true });
  } catch (e:any) {
    console.error(e);
    return res.status(500).json({ error: e.message });
  }
});

export default r;
