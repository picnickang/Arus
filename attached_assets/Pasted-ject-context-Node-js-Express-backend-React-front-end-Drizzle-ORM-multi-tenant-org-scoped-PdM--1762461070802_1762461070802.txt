Safer z-score

export function zScore(mu:number, sigma:number, x:number){
  const eps = Math.max(1e-6, 0.05*Math.abs(mu));
  const s = Math.max(Math.abs(sigma), eps);
  return (x-mu)/s;
}


bandRMS (bin-width aware)

export function bandRMS(freq:number[], mag:number[], bands:{lo:number;hi:number;name:string}[]){
  const n = Math.min(freq.length, mag.length);
  if(!n) return bands.map(b=>({name:b.name, value:0}));
  const df = new Array(n);
  for(let i=0;i<n;i++){
    const left = i?freq[i-1]:freq[i];
    const right = i<n-1?freq[i+1]:freq[i];
    df[i] = Math.max(0,(right-left)/2);
  }
  return bands.map(b=>{
    let energy=0;
    for(let i=0;i<n;i++){
      const f=freq[i]; if(f<b.lo||f>=b.hi) continue;
      const a=mag[i]||0; energy += (a*a)*df[i];
    }
    const width = Math.max(1e-9, b.hi-b.lo);
    return { name:b.name, value: Math.sqrt(energy/width) };
  });
}


Pearson pairwise

function pearsonPairwise(a:number[], b:number[]){
  let n=0,sumX=0,sumY=0,sumXX=0,sumYY=0,sumXY=0;
  const len=Math.min(a.length,b.length);
  for(let i=0;i<len;i++){
    const x=a[i], y=b[i];
    if(!Number.isFinite(x)||!Number.isFinite(y)) continue;
    n++; sumX+=x; sumY+=y; sumXX+=x*x; sumYY+=y*y; sumXY+=x*y;
  }
  if(n<2) return NaN;
  const cov=(sumXY-(sumX*sumY)/n)/n;
  const varX=(sumXX-(sumX*sumX)/n)/n;
  const varY=(sumYY-(sumY*sumY)/n)/n;
  const denom = Math.sqrt(Math.max(varX,0)*Math.max(varY,0))||0;
  return denom? cov/denom : NaN;
}
