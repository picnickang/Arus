Drop this in place of the broken export const sensorTemplates = pgTable("sensor_templates", { ... }); chunk:

export const sensorTemplates = pgTable(
  "sensor_templates",
  {
    id: varchar("id", { length: 36 })
      .primaryKey()
      .default(sql`gen_random_uuid()`),

    // Nullable: NULL = global “system” template (applies to all orgs)
    orgId: varchar("org_id")
      .references(() => organizations.id, { onDelete: "cascade" })
      .nullable(),

    // Short, stable identifier like "OIL-DEBRIS-GEARBOX"
    templateId: text("template_id").notNull(),

    // Human-readable name for UI
    name: text("name").notNull(),

    // Sensor type: oil_debris, vibration, pressure, temperature, etc.
    kind: text("kind").notNull(),

    // Measurement unit: e.g. "ppm", "mg/L", "°C", "g", etc.
    unit: text("unit").notNull(),

    // Compatible equipment types: ["gearbox", "engine", "thruster"]
    // Nullable so old rows/seeds without this field still work
    equipmentTypes: text("equipment_types").array().nullable(),

    // Arbitrary config for thresholds, sampling, transforms, etc.
    // Example:
    // {
    //   warn_threshold: 100,
    //   crit_threshold: 150,
    //   sample_rate_hz: 1,
    //   smoothing: { ema_alpha: 0.15 }
    // }
    fields: jsonb("fields").notNull(),

    // Free-form description / guidance
    notes: text("notes").nullable(),

    // True = baked-in system templates (protected),
    // False = org-created custom templates
    isSystemDefault: boolean("is_system_default")
      .notNull()
      .default(false),

    createdBy: text("created_by").nullable(),

    createdAt: timestamp("created_at", { mode: "date" })
      .notNull()
      .defaultNow(),

    updatedAt: timestamp("updated_at", { mode: "date" })
      .notNull()
      .defaultNow(),
  },
  (table) => ({
    // Fast lookups for “all system templates” and per-org + templateId
    orgTemplateIdx: index("sensor_templates_org_template_idx").on(
      table.orgId,
      table.templateId,
    ),
    kindIdx: index("sensor_templates_kind_idx").on(table.kind),
  }),
);
3. Zod schemas & TS types for templates



Add this right after the table definition:

// Select schema / type – what you read from DB
export const selectSensorTemplateSchema = createSelectSchema(sensorTemplates);
export type SensorTemplate = z.infer<typeof selectSensorTemplateSchema>;

// Insert/update schema / type – what APIs accept
export const insertSensorTemplateSchema = createInsertSchema(sensorTemplates, {
  // Optional runtime tweaks: ensure equipmentTypes is always array on input
  equipmentTypes: z
    .array(z.string())
    .optional()
    .nullable(),
  fields: z.record(z.any()), // explicit: JSON object
}).omit({
  id: true,
  createdAt: true,
  updatedAt: true,
  isSystemDefault: true, // system defaults set from backend/seed
  createdBy: true,
});

export type InsertSensorTemplate = z.infer<typeof insertSensorTemplateSchema>;
