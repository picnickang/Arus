type BucketRow = { ts: Date; features: Record<string, number> };
function bucketTelemetryRaw(
  rows: { timestamp: Date; features: Record<string, number> }[],
  stepMs = 1000
): BucketRow[] {
  const sorted = [...rows]
    .filter(r => r.timestamp && !isNaN(r.timestamp.getTime()))
    .sort((a,b) => a.timestamp.getTime() - b.timestamp.getTime());
  const buckets = new Map<number, Record<string, number>>();
  for (const r of sorted) {
    const t = Math.floor(r.timestamp.getTime()/stepMs)*stepMs;
    const acc = buckets.get(t) ?? {};
    for (const k of Object.keys(r.features)) {
      const v = (r.features as any)[k];
      if (Number.isFinite(v)) acc[k] = v;
    }
    buckets.set(t, acc);
  }
  return [...buckets.keys()].sort((a,b)=>a-b).map(t => ({ ts: new Date(t), features: buckets.get(t)! }));
}

async function getAdaptiveWeights(
  storage: IStorage, orgId: string, equipmentType: string
): Promise<{ lstm: number; rf: number; xgb: number }> {
  try {
    const lstm = await storage.getMlModels(orgId, 'lstm', 'active');
    const rf   = await storage.getMlModels(orgId, 'random_forest', 'active');
    const xgb  = await storage.getMlModels(orgId, 'xgboost', 'active');
    const pick = (arr:any[]) =>
      arr.filter(m => m.targetEquipmentType === equipmentType || m.targetEquipmentType == null)
         .sort((a,b) => ((b.performance?.f1Score ?? b.performance?.accuracy ?? 0) -
                         (a.performance?.f1Score ?? a.performance?.accuracy ?? 0)))[0];

    const l = pick(lstm); const r = pick(rf); const x = pick(xgb);
    if (!l && !r && !x) return STATIC_WEIGHTS[equipmentType] ?? STATIC_WEIGHTS.default;

    // Normalize recent performance (0.5..1.0 cap)
    const score = (m:any) => Math.max(0.5, Math.min(1.0, (m?.performance?.f1Score ?? m?.performance?.accuracy ?? 0)));
    const wl = l ? score(l) : 0;
    const wr = r ? score(r) : 0;
    const wx = x ? score(x) : 0;

    // gentle priors to keep stability similar to your original
    const bumpL = 1.05, bumpX = 1.1, bumpR = 1.0;
    let L = wl * bumpL, R = wr * bumpR, X = wx * bumpX;
    const sum = L+R+X || 1;
    return { lstm: L/sum, rf: R/sum, xgb: X/sum };
  } catch {
    return STATIC_WEIGHTS[equipmentType] ?? STATIC_WEIGHTS.default;
  }
}

function stats(vals: number[]) {
  const v = vals.filter(Number.isFinite);
  if (!v.length) return { avg: 0, max: 0, min: 0, std: 0 };
  const avg = v.reduce((a,b)=>a+b,0)/v.length;
  const max = Math.max(...v);
  const min = Math.min(...v);
  const variance = v.reduce((s,x)=>s+(x-avg)*(x-avg),0)/v.length;
  return { avg, max, min, std: Math.sqrt(variance) };
}
