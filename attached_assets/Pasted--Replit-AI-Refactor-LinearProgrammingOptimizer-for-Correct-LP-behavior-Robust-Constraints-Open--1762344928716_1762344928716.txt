Soft Lateness Penalty



Instead of forbidding late jobs, add a penalty term in the objective proportional to lateness (in days).



Add helper:

private latenessPenalty(job: MaintenanceJob, dayOffset: number, weights: OptimizationConstraints['priorityWeights']): number {
  if (!job.deadline) return 0;
  const dueOffset = Math.max(0, (Date.now() - job.deadline.getTime()) / (1000 * 60 * 60 * 24));
  const latenessDays = Math.max(0, dayOffset - dueOffset);
  const priorityWeight = this.getPriorityCost(job.priority, weights);
  // penalty grows faster for critical jobs
  return latenessDays * priorityWeight * 10;
}
