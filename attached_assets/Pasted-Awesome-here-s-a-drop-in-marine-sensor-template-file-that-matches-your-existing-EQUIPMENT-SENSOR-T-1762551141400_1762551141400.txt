// Optional helper if you want to merge into an existing template map
export function mergeMarineTemplates<T extends Record<string, TemplateEntry[]>>(base: T) {
  return { ...base, ...MARINE_EQUIPMENT_SENSOR_TEMPLATES };
}
Quick integration tip

If your EquipmentAnalyticsService.EQUIPMENT_SENSOR_TEMPLATES is private/readonly, just copy the blocks you want (e.g., main_engine, thruster_system, etc.) into that map.

If you control its creation, you can do:

import { MARINE_EQUIPMENT_SENSOR_TEMPLATES } from './marine-sensor-templates';
const EQUIPMENT_SENSOR_TEMPLATES = { 
  ...BASE_TEMPLATES, 
  ...MARINE_EQUIPMENT_SENSOR_TEMPLATES 
};
