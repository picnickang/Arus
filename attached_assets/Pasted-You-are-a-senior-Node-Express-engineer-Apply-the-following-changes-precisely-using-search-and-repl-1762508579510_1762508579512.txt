Step 3 — Safe wildcard CORS matching



Create server/utils/corsWildcard.ts:

export function wildcardToRegex(pat: string) {
  const escaped = pat.trim().replace(/[.+?^${}()|[\]\\]/g, '\\$&');
  const expanded = escaped.replace(/\\\*/g, '.*');
  return new RegExp(`^${expanded}$`);
}

export function originAllowed(origin: string, allowlist: string[]) {
  return allowlist.some(p => wildcardToRegex(p).test(origin));
}

Step 5 — Redacted + size-capped JSON response logging



Create server/utils/redact-log.ts:

const SENSITIVE_KEYS = new Set(['password','token','apikey','api_key','secret','authorization','hmac','access_token','refresh_token']);

function redactValue(v: any) {
  if (typeof v === 'string') {
    if (v.length <= 4) return '•redacted•';
    return `${v.slice(0, 2)}…•redacted•`;
  }
  return '•redacted•';
}

export function deepRedact(obj: any, depth = 0): any {
  if (depth > 5) return '[depth-capped]';
  if (!obj || typeof obj !== 'object') return obj;
  if (Array.isArray(obj)) return obj.slice(0, 50).map(v => deepRedact(v, depth + 1));

  const out: any = {};
  for (const [k, v] of Object.entries(obj)) {
    if (SENSITIVE_KEYS.has(k.toLowerCase())) out[k] = redactValue(v);
    else out[k] = deepRedact(v, depth + 1);
  }
  return out;
}

export function safeStringify(obj: any, maxLen = 2000) {
  try {
    const red = deepRedact(obj);
    let s = JSON.stringify(red);
    if (s.length > maxLen) s = s.slice(0, maxLen - 1) + '…';
    return s;
  } catch {
    return '[unserializable]';
  }
}

function withTimeout(p: Promise<any>, ms: number) {
  return Promise.race([p, new Promise((_r, rej) => setTimeout(() => rej(new Error('timeout')), ms))]);
}

const shutdown = async (sig: string) => {
  console.log(`\n↩ ${sig} received. Shutting down gracefully...`);
