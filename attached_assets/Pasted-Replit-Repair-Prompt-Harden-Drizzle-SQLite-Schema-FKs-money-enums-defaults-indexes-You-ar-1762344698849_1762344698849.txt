import { sqliteTable, text, integer, real, index, uniqueIndex, sql } from "drizzle-orm/sqlite-core";

// VESSELS
export const vessels = sqliteTable("vessels", {
  id: text("id").primaryKey(),
  orgId: text("org_id").notNull(),
  name: text("name").notNull(),
  vesselClass: text("vessel_class")
    .check(sql`vessel_class in ('tugboat','anchor_handling_tug','platform_supply_vessel','pilot_boat','work_boat','dynamic_positioning_dive','fast_ferry','landing_craft','seismic_surveillance','standby')`),
  condition: text("condition").notNull().default("good")
    .check(sql`condition in ('excellent','good','fair','poor','critical')`),
  onlineStatus: text("online_status").notNull().default("offline")
    .check(sql`online_status in ('online','offline')`),

  // MONEY: store in cents (integer)
  dayRateSgdCents: integer("day_rate_sgd_cents"), // new; backfill from old real if present

  // Counters as numbers
  operationDays: integer("operation_days").notNull().default(0),
  downtimeDays: integer("downtime_days").notNull().default(0),

  // Idempotency markers
  lastOperationIncrementDate: text("last_operation_increment_date"),
  lastDailyUpdateDate: text("last_daily_update_date"),

  createdAt: integer("created_at", { mode: "timestamp" }).default(sql`(unixepoch())`),
  updatedAt: integer("updated_at", { mode: "timestamp" }),
}, (t) => ({
  orgIdx: index("idx_vessels_org").on(t.orgId),
  nameIdx: index("idx_vessels_name").on(t.name),
}));
If you still have day_rate_sgd as real, keep it TEMPORARILY for migration. We will backfill and then drop it in a follow-up migration.
