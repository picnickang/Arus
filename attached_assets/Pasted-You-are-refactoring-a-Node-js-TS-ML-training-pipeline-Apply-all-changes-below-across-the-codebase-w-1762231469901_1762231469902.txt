export function chronoSplit<T extends { timestamp: Date }>(rows: T[], trainFrac = 0.8) {
  const sorted = [...rows].sort((a,b) => a.timestamp.getTime() - b.timestamp.getTime());
  const k = Math.max(1, Math.floor(sorted.length * trainFrac));
  return { train: sorted.slice(0, k), validation: sorted.slice(k) };
}

const SEED = 1337;
function seededShuffle<T>(arr: T[]) {
  let seed = SEED; const a = [...arr];
  for (let i = a.length - 1; i > 0; i--) {
    seed = (seed * 9301 + 49297) % 233280;
    const r = seed / 233280;
    const j = Math.floor(r * (i + 1));
    [a[i], a[j]] = [a[j], a[i]];
  }
  return a;
}

export function stratifiedSplit<T extends { label: string }>(rows: T[], trainFrac = 0.8) {
  const by = new Map<string, T[]>();
  for (const r of rows) by.set(r.label, [ ...(by.get(r.label) ?? []), r ]);
  const train: T[] = [], validation: T[] = [];
  for (const [, arr] of by) {
    const shuffled = seededShuffle(arr);
    const k = Math.max(1, Math.floor(shuffled.length * trainFrac));
    train.push(...shuffled.slice(0, k));
    validation.push(...shuffled.slice(k));
  }
  return { train, validation };
}

const labels = Array.from(new Set(validation.map(v => v.label)));

for (const sample of validation) {

  const pred = predictWithXGBoost(trainedModel, sample.features).prediction;
  for (const l of labels) {
    const s = perClass.get(l)!;
    if (pred === l && sample.label === l) s.tp++;
    else if (pred === l && sample.label !== l) s.fp++;
    else if (pred !== l && sample.label === l) s.fn++;

const f1s = [...perClass.values()].map(({tp,fp,fn}) => {
  const P = tp / Math.max(1, tp + fp);
  const R = tp / Math.max(1, tp + fn);
  return (P+R) ? (2*P*R)/(P+R) : 0;
});
const macroF1 = f1s.reduce((a,b)=>a+b,0) / Math.max(1, f1s.length);
