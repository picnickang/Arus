Step 2 – Fix Express static serving & SPA fallback

In server/index.ts (where the Express app is created and configured), locate the code that sets up static file serving for embedded/desktop mode. It currently logs:

“Checking: /Users/…/RecipeRealm/dist”

“Static file serving configured from: /Users/…/RecipeRealm/dist”

Make sure the code looks roughly like this (adapt to actual structure):

import fs from "fs";
import path from "path";
import express from "express";

const app = express();
const projectRoot = process.cwd();

const candidateStaticRoots = [
  path.join(projectRoot, "dist"),
  path.join(projectRoot, "client", "dist"),
];

let staticRoot: string | null = null;

for (const candidate of candidateStaticRoots) {
  if (fs.existsSync(candidate) && fs.existsSync(path.join(candidate, "index.html"))) {
    staticRoot = candidate;
    console.log("[Static] Using frontend build from:", candidate);
    break;
  } else if (fs.existsSync(candidate)) {
    console.warn("[Static] Found directory", candidate, "but no index.html inside");
  }
}

if (!staticRoot) {
  console.warn("[Static] No valid frontend build directory found - SPA UI will not be served");
} else {
  app.use(express.static(staticRoot));

  // SPA fallback route
  app.get("*", (req, res, next) => {
    if (req.path.startsWith("/api/")) {
      return next();
    }
    res.sendFile(path.join(staticRoot!, "index.html"));
  });
}
