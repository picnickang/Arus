// calculateStockStatus guard
function calculateStockStatus(onHand: number, reserved: number, minStock: number, maxStock: number) {
  const available = Math.max(0, onHand - reserved);
  if (available <= 0) return 'critical';
  if (minStock <= 0) return available > 0 ? 'adequate' : 'critical';
  if (available < Math.max(1, Math.floor(minStock * 0.5))) return 'critical';
  if (available < minStock) return 'low';
  if (maxStock > 0 && available > Math.ceil(maxStock * 1.2)) return 'excess';
  return 'adequate';
}
