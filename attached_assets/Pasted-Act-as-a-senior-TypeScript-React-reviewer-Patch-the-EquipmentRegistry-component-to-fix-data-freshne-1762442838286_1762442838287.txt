export async function processPendingEvents(limit = 100): Promise<number> {
  try {
    // 1) Claim
    const claimed = await db.transaction(async (tx) => {
      const rows = await tx
        .select()
        .from(syncOutbox)
        .where(and(
          eq(syncOutbox.processed, false),
          sql`${syncOutbox.processing} = false`,
          sql`${syncOutbox.processingAttempts} < 5`,
          sql`${syncOutbox.nextAttemptAt} IS NULL OR ${syncOutbox.nextAttemptAt} <= NOW()`
        ))
        .orderBy(syncOutbox.createdAt)
        .limit(limit);

      if (!rows.length) return rows;

      // mark claimed
      await Promise.all(rows.map(r =>
        tx.update(syncOutbox)
          .set({ processing: true, processingStartedAt: new Date() })
          .where(eq(syncOutbox.id, r.id))
      ));
      return rows;
    });

    let processedCount = 0;
    for (const ev of claimed) {
      try {
        syncEventBus.emit(ev.eventType as EventType, ev.payload);
        await db.update(syncOutbox)
          .set({ processed: true, processedAt: new Date(), processing: false, processingAttempts: ev.processingAttempts + 1 })
          .where(eq(syncOutbox.id, ev.id));
        processedCount++;
      } catch (eventErr) {
        console.error(`[SyncEvents] Failed to process event ${ev.id}: ${errMsg(eventErr)}`);
        const attempts = ev.processingAttempts + 1;
        const backoffMs = Math.min(60_000, 2 ** attempts * 1000); // cap at 60s
        await db.update(syncOutbox)
          .set({
            processing: false,
            processingAttempts: attempts,
            nextAttemptAt: new Date(Date.now() + backoffMs),
            status: attempts >= 5 ? 'failed' : 'pending'
          })
          .where(eq(syncOutbox.id, ev.id));
      }
    }
    return processedCount;
  } catch (e) {
    console.error(`[SyncEvents] Failed to process pending events: ${errMsg(e)}`);
    return 0;
  }
}

export interface PersistedMessage {
  id?: string;
  topic: string;
  payload: any;
  qos: 0 | 1 | 2;
  retain: boolean;
  createdAt?: Date;
}

export class QueueManager {
  constructor(
    private queueDir: string,
    private maxQueue: number
  ) {}

  async init() {
    await fs.mkdir(this.queueDir, { recursive: true }).catch(() => {});
  }

  async size(): Promise<number> {
    const [{ count }] = await db.select({ count: sql<number>`COUNT(*)` }).from(syncOutboxSqlite).where(eq(syncOutboxSqlite.processed, 0));
    return Number(count || 0);
  }

  async enqueue(msg: PersistedMessage) {
    const current = await this.size();
    if (current >= this.maxQueue) {
      // Drop oldest
      const oldest = await db.select().from(syncOutboxSqlite).where(eq(syncOutboxSqlite.processed, 0)).orderBy(syncOutboxSqlite.createdAt).limit(1);
      if (oldest[0]) {
        await db.delete(syncOutboxSqlite).where(eq(syncOutboxSqlite.id, oldest[0].id));
      }
    }
    await db.insert(syncOutboxSqlite).values({
      id: cryptoRandomId(),
      eventType: msg.topic, // reuse column
      payload: JSON.stringify(msg),
      processed: 0,
      processingAttempts: 0,
      createdAt: new Date(),
    });
  }

  async takeBatch(limit: number) {
    // Fetch unprocessed batch FIFO
    const rows = await db.select().from(syncOutboxSqlite)
      .where(eq(syncOutboxSqlite.processed, 0))
      .orderBy(syncOutboxSqlite.createdAt)
      .limit(limit);
    return rows.map(r => ({ dbId: r.id, ...JSON.parse(r.payload as string) as PersistedMessage }));
  }

  async markProcessed(dbId: string) {
    await db.update(syncOutboxSqlite)
      .set({ processed: 1, processedAt: new Date(), processingAttempts: sql`${syncOutboxSqlite.processingAttempts} + 1` })
      .where(eq(syncOutboxSqlite.id, dbId));
  }

  async markFailed(dbId: string) {
    await db.update(syncOutboxSqlite)
      .set({ processingAttempts: sql`${syncOutboxSqlite.processingAttempts} + 1` })
      .where(eq(syncOutboxSqlite.id, dbId));
  }
}

// server/sync/Deduplicator.ts
export class Deduplicator {
  private cache = new Map<string, number>();
  constructor(private ttlMs = 10 * 60 * 1000, private maxEntries = 50000) {}

  seen(id: string): boolean {
    const now = Date.now();
    // cleanup occasionally
    if (this.cache.size > this.maxEntries) {
      for (const [k, v] of this.cache) {
        if (now - v > this.ttlMs) this.cache.delete(k);
      }
    }
    const t = this.cache.get(id);
    if (t && now - t < this.ttlMs) return true;
    this.cache.set(id, now);
    return false;
  }
}

function limiter(max: number) {
  let active = 0, queue: any[] = [];
  const next = () => {
    if (active >= max || queue.length === 0) return;
    active++;
    const fn = queue.shift();
    fn().finally(() => { active--; next(); });
  };

    queue.push(() => fn().then(res, rej));
    next();
  });
}

src/realtime/schemas.ts

import { z } from 'zod';

export const DataChangeMessageSchema = z.object({
  type: z.literal('data_change'),
  entity: z.string().min(1),
  operation: z.enum(['create', 'update', 'delete']),
  data: z.unknown(),
  timestamp: z.string().or(z.date().transform(d => new Date(d).toISOString())),
});

export type DataChangeMessage<T = unknown> = z.infer<typeof DataChangeMessageSchema> & { data: T };

export const MqttEnvelopeSchema = z.object({
  // Sometimes we wrap payloads when using retained messages or envelopes
  type: z.string().optional(),
  entity: z.string().optional(),
  operation: z.enum(['create', 'update', 'delete']).optional(),
  data: z.unknown().optional(),
  timestamp: z.string().optional(),
  // Fallback passthrough for server-originated normalized messages
  payload: z.unknown().optional(),
});

src/realtime/mqttClient.ts

import { DataChangeMessageSchema, type DataChangeMessage, MqttEnvelopeSchema } from './schemas';
import type { IClientOptions, MqttClient } from 'mqtt';
import mqtt from 'mqtt/dist/mqtt'; // browser build

type OnMessage = (msg: DataChangeMessage) => void;

export interface MqttRealtimeOptions {
  url?: string; // e.g. wss://broker.example.com:8083/mqtt
  clientId?: string;
  qos?: 0 | 1 | 2;
  topics?: string[]; // e.g. ['vessel/sync/+/events', 'vessel/sync/+/state']
  reconnectPeriodMs?: number;
  enabled?: boolean;
}

export class MqttRealtime {
  private client: MqttClient | null = null;
  private onMessage?: OnMessage;
  private options: Required<MqttRealtimeOptions>;
  private hasReconnectedOnce = false;

  constructor(opts: MqttRealtimeOptions = {}) {
    const url = opts.url ?? import.meta.env.VITE_MQTT_URL;
    this.options = {
      url: url || '',
      clientId: opts.clientId ?? `web_${crypto.randomUUID()}`,
      qos: opts.qos ?? 1,
      topics: opts.topics ?? ['vessel/sync/+/events', 'vessel/sync/+/state'],
      reconnectPeriodMs: opts.reconnectPeriodMs ?? 5000,
      enabled: Boolean(opts.enabled ?? !!url),
    };
  }

  start(onMessage: OnMessage) {
    this.onMessage = onMessage;
    if (!this.options.enabled || !this.options.url) return;

    const connectOptions: IClientOptions = {
      clientId: this.options.clientId,
      clean: true,
      reconnectPeriod: this.options.reconnectPeriodMs,
      keepalive: 60,
    };

    try {
      this.client = mqtt.connect(this.options.url, connectOptions);

      this.client.on('connect', () => {
        // Subscribe after (re)connect
        this.options.topics.forEach((t) => {
          this.client?.subscribe(t, { qos: this.options.qos }, (err) => {
            if (err) console.error('[MQTT] subscribe error', t, err);
          });
        });

        // Mark reconnection
        if (this.hasReconnectedOnce && this.onReconnect) {
          this.onReconnect();
        }
        this.hasReconnectedOnce = true;
        console.info('[MQTT] connected:', this.options.url);
      });

      this.client.on('reconnect', () => {
        console.info('[MQTT] reconnecting…');
      });

      this.client.on('error', (err) => {
        console.error('[MQTT] error:', err?.message || err);
      });

      this.client.on('message', (_topic, payload) => {
        // Payload may be DataChangeMessage or wrapped envelope; normalize
        try {
          const raw = JSON.parse(payload.toString());
          // Some brokers/publishers send an envelope; try unwrap
          const env = MqttEnvelopeSchema.safeParse(raw);
          const base = env.success && env.data.payload ? env.data.payload : raw;

          const parsed = DataChangeMessageSchema.safeParse(base);
          if (parsed.success) {
            onMessage(parsed.data);
          } else {
            // Ignore malformed but log once
            // console.debug('[MQTT] ignored non-data_change message', raw);
          }
        } catch {
          // ignore unparseable payloads
        }
      });
    } catch (e) {
      console.warn('[MQTT] failed to initialize client; continuing without MQTT', e);
    }
  }

  stop() {
    try {
      this.client?.end(true);
    } catch {}
    this.client = null;
  }

  // Hook can set this to trigger global refetch on reconnect
  onReconnect?: () => void;
}

src/realtime/useRealtimeSync.ts

import { useEffect, useMemo, useRef } from 'react';
import { queryClient } from '@/lib/queryClient';
import { ENTITY_QUERY_MAP } from './syncEntityMap';
import { DataChangeMessageSchema, type DataChangeMessage } from './schemas';
import { MqttRealtime } from './mqttClient';
import { useWebSocket } from './useWebSocket';

type BatchKey = string; // entity name
const DEBOUNCE_MS = 250;

export function useRealtimeSync() {
  // Existing WS hook (assumed provided)
  const { lastMessage, subscribe, isConnected: wsConnected } = useWebSocket({ autoConnect: true });

  // MQTT client (browser)
  const mqtt = useMemo(() => new MqttRealtime({
    // VITE_MQTT_URL is optional; if absent, MQTT stays disabled
    enabled: true, // will no-op if no URL
  }), []);

  const debounceTimer = useRef<ReturnType<typeof setTimeout> | null>(null);
  const pendingEntities = useRef<Set<BatchKey>>(new Set());
  const revalidatedAfterReconnect = useRef({ ws: false, mqtt: false });

  // --- Helpers ---
  const scheduleBatch = () => {
    if (debounceTimer.current) clearTimeout(debounceTimer.current);
    debounceTimer.current = setTimeout(() => {
      const entities = Array.from(pendingEntities.current);
      pendingEntities.current.clear();

      // Collect unique keys
      const keys = new Set<string>();
      entities.forEach((e) => (ENTITY_QUERY_MAP[e] || []).forEach(k => keys.add(k)));

      const start = performance.now();
      keys.forEach((key) => {
        queryClient.invalidateQueries({ queryKey: [key] });
      });
      const elapsed = Math.round(performance.now() - start);

      if (keys.size > 0) {
        console.info(`[Realtime] Invalidated ${keys.size} keys from ${entities.length} entities in ${elapsed}ms`, {
          entities,
          keys: Array.from(keys),
        });
      }
    }, DEBOUNCE_MS);
  };

  const handleMessage = (msg: unknown) => {
    const parsed = DataChangeMessageSchema.safeParse(msg);
    if (!parsed.success) return;
    const message = parsed.data as DataChangeMessage;

    // Track entity for batched invalidation
    pendingEntities.current.add(message.entity);
    scheduleBatch();
  };

  // --- WebSocket wiring ---
  useEffect(() => {
    if (wsConnected) {
      subscribe('data:all');
      if (!revalidatedAfterReconnect.current.ws) {
        // One-time global revalidation on first connect
        queryClient.invalidateQueries();
        revalidatedAfterReconnect.current.ws = true;
        console.info('[Realtime] WS reconnected → global refetch triggered once');
      }
    }
  }, [wsConnected, subscribe]);

  useEffect(() => {
    if (!lastMessage) return;
    handleMessage(lastMessage);
  }, [lastMessage]);

  // --- MQTT wiring ---
  useEffect(() => {
    mqtt.onReconnect = () => {
      if (!revalidatedAfterReconnect.current.mqtt) {
        queryClient.invalidateQueries();
        revalidatedAfterReconnect.current.mqtt = true;
        console.info('[Realtime] MQTT reconnected → global refetch triggered once');
      }
    };
    mqtt.start((msg) => handleMessage(msg));
    return () => mqtt.stop();
  }, [mqtt]);

  return {
    isConnected: wsConnected, // for UI; MQTT is internal resilience layer
  };
}

Acceptance Criteria

When either WebSocket or MQTT delivers a valid data_change message:

It is validated with Zod and normalized.

The corresponding entity is added to a debounced batch, and affected query keys from ENTITY_QUERY_MAP are invalidated once per batch.

Upon first reconnect (WS or MQTT), a single global refetch runs, not repeatedly.

If VITE_MQTT_URL is absent, app works with WebSocket only (no errors).

Logs show concise batch summaries (entities, keys, elapsed ms).

No TypeScript any leaks; build is clean.

Quick Test Plan

MQTT only: temporarily disable WS, run a local broker (e.g., eclipse-mosquitto), publish JSON to vessel/sync/work_orders/events. Verify invalidations.

Malformed payloads: Publish junk or missing fields — ensure the app ignores them without errors.

Notes / Gotchas

Brokers often require a WebSocket listener (e.g., port 8083). Configure Mosquitto or EMQX accordingly.

If you later add retained state topics, ensure your backend normalizes them to the same DataChangeMessage shape (or keep the envelope and unwrap like shown).

Keep entity names consistent across backend & frontend to leverage ENTITY_QUERY_MAP.
