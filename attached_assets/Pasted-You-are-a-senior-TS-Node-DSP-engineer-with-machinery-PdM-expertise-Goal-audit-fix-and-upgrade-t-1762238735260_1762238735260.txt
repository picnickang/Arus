A) Single-sided spectrum with Hann window & scaling

// dsp/utils.ts
export function applyWindow(x: number[], type: 'hann'|'rect' = 'hann'): number[] {
  if (type === 'rect') return x.slice();
  const N = x.length;
  const w = x.map((_, n) => 0.5 * (1 - Math.cos((2*Math.PI*n)/(N-1))));
  return x.map((v, i) => v * w[i]);
}

export function singleSidedSpectrum(x: number[], sampleRate: number, window: 'hann'|'rect'='hann') {
  const N = x.length;
  const xw = applyWindow(x, window);
  // @ts-ignore fft-js types
  const X = FFTJS.fft(xw); // [[re,im], ...]
  const mag: number[] = [];
  const K = Math.floor(N/2);
  const cg = window === 'hann' ? 0.5 : 1;           // coherent gain
  const scale = 1/(cg * N);
  for (let i=0;i<=K;i++){
    const re = X[i][0], im = X[i][1];
    let m = Math.sqrt(re*re + im*im) * scale;
    if (i>0 && i<K) m *= 2; // single-sided scaling
    mag.push(m);
  }
  const freq = mag.map((_, i) => i * sampleRate / N);
  return { freq, mag };
}
B) Sample-rate inference from timestamps

export function inferSampleRate(samples: {ts: Date}[], fallback=1000): number {
  if (!samples?.length || samples.length < 4) return fallback;
  const dts = [];
  for (let i=1;i<samples.length;i++){
    const dt = (samples[i].ts.getTime() - samples[i-1].ts.getTime())/1000;
    if (dt>0) dts.push(dt);
  }
  if (dts.length < 3) return fallback;
  dts.sort((a,b)=>a-b);
  const med = dts[Math.floor(dts.length/2)];
  if (!isFinite(med) || med<=0) return fallback;
  const sr = 1/med;
  // guard for silly spikes
  return Math.max(1, Math.min(100_000, sr));
}
C) Harmonics with relative tolerance & peaks

export function extractHarmonics(freq: number[], mag: number[], f0: number, relTol=0.05, maxH=10){
  if (!f0 || f0 <= 0) return [];
  const peaks = peakPick(freq, mag, { prominence: median(mag)*3, minSepHz: freq[1]-freq[0] });
  const out: {freq:number; magnitude:number}[] = [];
  for (let h=2; h<=maxH; h++){
    const target = f0*h;
    const tol = target*relTol;
    const cand = peaks
      .filter(p => Math.abs(p.freq-target) <= tol)
      .sort((a,b)=>b.magnitude-a.magnitude)[0];
    if (cand) out.push({ freq: cand.freq, magnitude: cand.magnitude });
  }
  return out.slice(0, 5);
}
D) Shaft frequency handling

function estimateShaftHz(freq:number[], mag:number[], maxHz=100){
  const idxs = freq.map((f,i)=> [i,f]).filter(([_,f])=> f>0 && f<=maxHz).map(([i])=> i);
  if (idxs.length===0) return null;
  const iMax = idxs.reduce((best,i)=> mag[i] > mag[best] ? i : best, idxs[0]);
  return freq[iMax] || null;
}
E) ISO band energy with Nyquist clip

function bandRms(freq:number[], mag:number[], lo:number, hi:number){
  const nyq = freq[freq.length-1];
  const minF = Math.max(0, lo), maxF = Math.min(hi, nyq);
  if (minF >= maxF) return 0;
  let sum=0, n=0;
  for (let i=0;i<freq.length;i++){
    if (freq[i] >= minF && freq[i] <= maxF){ sum += mag[i]*mag[i]; n++; }
  }
  return n>0 ? Math.sqrt(sum/n) : 0;
}
