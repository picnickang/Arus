1) Make adminQueryFn use React-Query’s context and robust URL building



Replace adminQueryFn with:

function adminQueryFn() {
  return async ({ queryKey }: { queryKey: (string | number)[] }) => {
    const sessionToken = getAdminSessionToken();
    if (!sessionToken) throw new Error('Admin session not active. Please unlock admin mode first.');

    // Accept either a single path element like "/api/admin/settings"
    // or segments like ["/api/admin/patches", "history"]
    const path = queryKey.join('/').replace(/\/+/g, '/'); // normalize double slashes

    const headers: Record<string, string> = {
      Authorization: `Bearer ${sessionToken}`,
      'x-org-id': 'default-org-id',
    };

    const res = await fetch(path, { headers, credentials: 'include' });
    if (!res.ok) {
      const text = await res.text();
      if (res.status === 401) throw new Error('Admin session expired. Please unlock admin mode again.');
      throw new Error(`${res.status}: ${text}`);
    }
    return res.status === 204 ? null : res.json();
  };
}

function coerceValueByType(raw: string, dataType: SystemSettingForm['dataType']) {
  try {
    switch (dataType) {
      case 'number':  return Number(raw);
      case 'boolean': return /^(true|1|yes)$/i.test(raw);
      case 'object':  return raw ? JSON.parse(raw) : {};
      case 'array':   return raw ? JSON.parse(raw) : [];
      case 'string':
      default:        return String(raw ?? '');
    }
  } catch {
    // Fallback to string if parsing fails; caller can toast error if desired
    return raw;
  }
}
