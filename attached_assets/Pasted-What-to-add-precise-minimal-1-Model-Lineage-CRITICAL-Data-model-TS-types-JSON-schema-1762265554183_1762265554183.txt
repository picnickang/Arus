// apps/server/src/governance/lineage.ts
import { createHash } from "crypto";
import fs from "fs/promises";
import path from "node:path";
import type { LineageRecord } from "./types";

const LINEAGE_FILE = process.env.LINEAGE_FILE ?? "./checkpoints/lineage.jsonl";

export async function sha256File(p: string) {
  const h = createHash("sha256");
  h.update(await fs.readFile(p));
  return h.digest("hex");
}

export async function appendLineage(rec: LineageRecord) {
  await fs.mkdir(path.dirname(LINEAGE_FILE), { recursive: true });
  await fs.appendFile(LINEAGE_FILE, JSON.stringify(rec) + "\n", "utf8");
}

export async function recordTraining(opts: Omit<LineageRecord, "predictionCount">) {
  const r: LineageRecord = { ...opts, predictionCount: 0 };
  await appendLineage(r);
  return r;
}

export async function incrementPredCount(modelId: string) {
  // simple: append an “update” delta; or keep a KV store. Appending is audit-friendly.
  const delta = { type: "predCount", modelId, ts: new Date().toISOString(), inc: 1 };
  await fs.appendFile(LINEAGE_FILE, JSON.stringify(delta) + "\n");
}

Chain hashing

// apps/server/src/governance/provenance.ts
import { createHash } from "crypto";
import fs from "fs/promises";
import path from "node:path";

const PROV_FILE = process.env.PROVENANCE_FILE ?? "./checkpoints/provenance.jsonl";

export async function appendProvenance(event: Record<string, unknown>) {
  await fs.mkdir(path.dirname(PROV_FILE), { recursive: true });
  const prev = await lastHash();
  const payload = { ...event, ts: new Date().toISOString(), prevHash: prev ?? null };
  const hash = sha256(JSON.stringify(payload));
  const row = { ...payload, hash };
  await fs.appendFile(PROV_FILE, JSON.stringify(row) + "\n", "utf8");
  return row;
}

function sha256(s: string) {
  return createHash("sha256").update(s).digest("hex");
}

async function lastHash() {
  try {
    const text = await fs.readFile(PROV_FILE, "utf8");
    const last = text.trim().split("\n").pop();
    if (!last) return null;
    return JSON.parse(last).hash as string;
  } catch { return null; }
}

// apps/server/scripts/verify_provenance.ts
import fs from "fs";

const file = process.env.PROVENANCE_FILE ?? "./checkpoints/provenance.jsonl";
const rows = fs.readFileSync(file, "utf8").trim().split("\n").map(JSON.parse);

function sha256(str: string) {
  const { createHash } = require("crypto");
  return createHash("sha256").update(str).digest("hex");
}

let ok = true, prev: string | null = null;
for (const r of rows) {
  const copy = { ...r }; const hash = copy.hash; delete copy.hash;
  const calc = sha256(JSON.stringify(copy));
  if (calc !== hash) { ok = false; console.error("Hash mismatch", r); break; }
  if (copy.prevHash !== (prev ?? null)) { ok = false; console.error("Chain break", r); break; }
  prev = hash;
}
