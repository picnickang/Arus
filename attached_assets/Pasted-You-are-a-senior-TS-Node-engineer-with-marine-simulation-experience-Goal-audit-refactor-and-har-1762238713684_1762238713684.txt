// Canonical units: speed(kn), distance(NM), fuel(t/h), power(kW)
export const FUEL_DENSITY_T_PER_L = 0.00084; // 0.84 t/m³ → 0.00084 t/L (configurable)

export function lphToTph(lph: number, density = FUEL_DENSITY_T_PER_L): number {
  return lph * density;
}
export function tphToLph(tph: number, density = FUEL_DENSITY_T_PER_L): number {
  return tph / density;
}

export function clamp(v: number, lo: number, hi: number): number {
  return Math.min(hi, Math.max(lo, v));
}

export function boundedSpeed(kn: number, maxKn: number): number {
  return clamp(kn, 0, Math.max(0, maxKn));
}

E) Fuel Consumption Correctness



Replace calculateFuelConsumption():

private calculateFuelConsumption(state: TwinState, physics: PhysicsModel, specs?: VesselSpecifications): { lph: number; tph: number } {
  let totalLph = 0;
  const basePower = specs?.enginePower ?? physics.machinery.mainEngines[0]?.power ?? 0;
  for (const engine of Object.values(state.machinery.engines)) {
    const eff = engine.efficiency ?? 0.42;
    const powerUsed = basePower * engine.load;           // kW
    const lph = powerUsed * physics.propulsion.fuelConsumption; // L/h per kW
    totalLph += lph / Math.max(eff, 0.1); // penalize poor efficiency
  }
  const tph = lphToTph(totalLph);
  return { lph: totalLph, tph };
}
Then keep state.fuel.consumptionRate in t/h and present t/day only when needed.
