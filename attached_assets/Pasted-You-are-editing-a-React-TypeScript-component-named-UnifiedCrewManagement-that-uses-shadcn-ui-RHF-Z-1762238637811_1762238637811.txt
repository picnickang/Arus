Lightweight debounce for search

Add a small hook near the top of the component (or outside if you prefer):

function useDebouncedValue<T>(value: T, delay = 250) {
  const [debounced, setDebounced] = useState(value);
  useEffect(() => {
    const t = setTimeout(() => setDebounced(value), delay);
    return () => clearTimeout(t);
  }, [value, delay]);
  return debounced;
}

  onMutate: async (crewId: string) => {
    await queryClient.cancelQueries({ queryKey: ['/api/crew'] });
    const prev = queryClient.getQueryData<Crew[]>(['/api/crew']);
