private determineCostTrend(ordersWithCost: WorkOrder[]): 'increasing' | 'decreasing' | 'stable' {
  if (ordersWithCost.length < 4) return 'stable';
  const sorted = [...ordersWithCost].sort((a,b)=>+new Date(a.createdAt) - +new Date(b.createdAt));
  const q = Math.max(1, Math.floor(sorted.length / 4));
  const firstQuarter = sorted.slice(0, q);
  const lastQuarter  = sorted.slice(-q);

  const avg = (arr: WorkOrder[]) => arr.length
    ? arr.reduce((s, wo) => s + (wo.estimatedCost || 0), 0) / arr.length
    : 0;

  const avgFirst = avg(firstQuarter);
  const avgLast  = avg(lastQuarter);

  if (avgFirst === 0) return avgLast === 0 ? 'stable' : 'increasing';

  const change = ((avgLast - avgFirst) / avgFirst) * 100;
  if (Math.abs(change) < 15) return 'stable';
  return change > 0 ? 'increasing' : 'decreasing';
}

private calculateEfficiencyTrends(telemetry: EquipmentTelemetry[]): VesselLearnings['operationalInsights']['efficiencyTrends'] {
  const trends: VesselLearnings['operationalInsights']['efficiencyTrends'] = [];
  const sensorTypes = [...new Set(telemetry.map(t => t.sensorType).filter(Boolean))];

  sensorTypes.forEach(sensorType => {
    const readings = telemetry
      .filter(t => t.sensorType === sensorType)
      .sort((a, b) => +new Date(a.ts) - +new Date(b.ts));

    if (readings.length >= 10) {
      const mid = Math.floor(readings.length / 2);
      const firstHalf = readings.slice(0, mid);
      const secondHalf = readings.slice(mid);

      const avg = (arr: typeof readings) => arr.reduce((s, r) => s + r.value, 0) / (arr.length || 1);
      const avgFirst = avg(firstHalf);
      const avgSecond = avg(secondHalf);

      const change = avgFirst === 0
        ? (avgSecond > 0 ? 100 : 0)
        : ((avgSecond - avgFirst) / Math.abs(avgFirst)) * 100;

      let trend: 'improving' | 'declining' | 'stable' = 'stable';
      if (Math.abs(change) > 10) trend = change > 0 ? 'improving' : 'declining';

      trends.push({ metric: sensorType, trend, change: Math.round(change * 10) / 10 });
    }
  });

  return trends;
}

8) Predictive lead time — safe windows & baselines



Replace calculatePredictiveLeadTime with:

private calculatePredictiveLeadTime(readings: EquipmentTelemetry[], failures: WorkOrder(): number {
  if (failures.length === 0 || readings.length < 10) return 0;

  const leadTimes: number[] = [];
  const WINDOW_MS = 168 * 60 * 60 * 1000; // 168h

  failures.forEach(failure => {
    const failureTime = +new Date(failure.createdAt);
    const window = readings
      .filter(r => {
        const t = +new Date(r.ts);
        return t < failureTime && t > failureTime - WINDOW_MS;
      })
      .sort((a, b) => +new Date(b.ts) - +new Date(a.ts));

    if (window.length >= 5) {
      const recent = window.slice(0, Math.min(5, window.length));
      const baselineTail = window.slice(-Math.min(10, window.length));

      const recentAvg = recent.reduce((s, r) => s + r.value, 0) / recent.length;
      const normalAvg = baselineTail.reduce((s, r) => s + r.value, 0) / baselineTail.length;

      if (!normalAvg) return; // skip; cannot compute ratio

      if (Math.abs(recentAvg - normalAvg) / Math.abs(normalAvg) > 0.2) {
        const anomalyTime = +new Date(recent[0].ts);
        const hoursLead = (failureTime - anomalyTime) / (60 * 60 * 1000);
        if (hoursLead > 0) leadTimes.push(hoursLead);
      }
    }
  });

  if (!leadTimes.length) return 0;
  return Math.round(leadTimes.reduce((s, t) => s + t, 0) / leadTimes.length);
}

private calculateAverageResolutionTime(workOrders: WorkOrder[]): number {
  const completed = workOrders.filter(wo => wo.status === 'completed' && wo.completedAt);
  if (!completed.length) return 0;

  const totalHours = completed.reduce((sum, wo) => {
    const start = +new Date(wo.createdAt);
    const end = +new Date(wo.completedAt!);
    return sum + Math.max(0, (end - start) / (60 * 60 * 1000));
  }, 0);

  return +((totalHours / completed.length)).toFixed(1);
}
