export const kbDocs = pgTable('kb_docs', {

  orgId: text('org_id'),
  name: text('name').notNull(),
  source: text('source'),

export const kbChunks = pgTable('kb_chunks', {

3) Embedding Provider (Local-first, OpenAI fallback)



Create server/rag/embeddings.ts:

import * as path from "path";

let provider = process.env.EMBEDDINGS_PROVIDER || "local"; // "local" | "openai"
const dims = provider === "openai" ? 1536 : 384;

export const EMBEDDING_DIMS = dims;

export async function embedBatch(texts: string[]): Promise<number[][]> {
  if (provider === "openai" && process.env.OPENAI_API_KEY) {
    const { OpenAIEmbeddings } = await import("langchain/embeddings/openai");
    const embedder = new OpenAIEmbeddings({ modelName: "text-embedding-3-small" });
    return await embedder.embedDocuments(texts);
  } else {
    // Local open-source embeddings
    const { pipeline } = await import("@xenova/transformers");
    const embed = await pipeline("feature-extraction", "Xenova/all-MiniLM-L6-v2");
    const out: number[][] = [];
    for (const t of texts) {
      const result: any = await embed(t, { pooling: "mean", normalize: true });
      out.push(Array.from(result.data));
    }
    return out;
  }
}
4) OCR + PDF Extraction + Chunking



Create server/rag/ingest.ts:

import fs from "fs/promises";
import pdfParse from "pdf-parse";
import Tesseract from "tesseract.js";
import sharp from "sharp";
import path from "path";
import { db } from "../db";
import { kbDocs, kbChunks } from "@shared/schema";
import { embedBatch, EMBEDDING_DIMS } from "./embeddings";
import { sql } from "drizzle-orm";

// Simple sentence-aware chunker ~800–1200 chars
function chunkText(text: string, target = 1000): string[] {
  const sents = text.match(/[^.!?]+[.!?]\s*/g) || [text];
  const chunks: string[] = [];
  let buf = "";
  for (const s of sents) {
    if ((buf + s).length <= target) buf += s;
    else { if (buf.trim()) chunks.push(buf.trim()); buf = s; }
  }
  if (buf.trim()) chunks.push(buf.trim());
  return chunks;
}

// Image OCR helper (preprocess for better OCR)
async function ocrImage(filePath: string): Promise<string> {
  const pre = await sharp(filePath)
    .grayscale()
    .normalize()
    .sharpen()
    .toBuffer();
  const { data: { text } } = await Tesseract.recognize(pre, 'eng', { logger: () => {} });
  return (text || "").trim();
}

export async function extractText(filePath: string): Promise<{ text: string; meta: any; }> {
  const ext = path.extname(filePath).toLowerCase();
  const base = path.basename(filePath);
  let text = "";
  const meta: any = { pages: 0, source: base };

  try {
    if (ext === ".pdf") {
      const buf = await fs.readFile(filePath);
      const parsed = await pdfParse(buf);
      text = (parsed.text || "").trim();
      meta.pages = parsed.numpages || 0;
      // If PDF had no extractable text, try OCR each page fallback (optional here)
      if (!text) {
        // (Lightweight approach: export first page as image for OCR or require user to upload image scans)
        // Keep simple for now.
      }
    } else if ([".png", ".jpg", ".jpeg", ".tif", ".tiff", ".bmp"].includes(ext)) {
      text = await ocrImage(filePath);
      meta.pages = 1;
    }
  } catch (e) {
    console.error("Ingestion error:", e);
  }
  return { text, meta };
}

export async function ingestFileToKb(filePath: string, orgId: string | null) {
  const { text, meta } = await extractText(filePath);
  if (!text) {
    console.warn(`[RAG] No text extracted from: ${filePath}`);
    return { createdDocId: null, chunks: 0 };
  }
  const parts = chunkText(text, 1000);
  const [doc] = await db.insert(kbDocs).values({
    orgId,
    name: meta?.source || path.basename(filePath),
    source: filePath,
    numChunks: parts.length,
    metadata: meta || {},
  }).returning({ id: kbDocs.id });

  // Batch embed for speed
  const embeddings = await embedBatch(parts);
  // Insert chunks
  for (let i = 0; i < parts.length; i++) {
    await db.insert(kbChunks).values({
      docId: doc.id,
      text: parts[i],
      embedding: embeddings[i] as any,
      ord: i,
    });
  }
  console.log(`📘 Ingested ${filePath} → ${parts.length} chunks`);
  return { createdDocId: doc.id, chunks: parts.length };
}

// Simple vector search
export async function searchKb(orgId: string | null, queryEmbedding: number[], k = 5) {
  // If org scoping matters for chunks, join via docs
  return await db.execute(sql`
    SELECT c.text, c.ord, d.name, d.id as doc_id
    FROM kb_chunks c
    JOIN kb_docs d ON (c.doc_id = d.id)
    WHERE (${orgId}::text IS NULL OR d.org_id = ${orgId})
    ORDER BY c.embedding <-> ${queryEmbedding}::vector
    LIMIT ${k};
  `);
}
5) Oil-Report Parser (regex from OCR text)



Create server/rag/oil-report.ts:

export interface OilReport {
  viscosity_cSt_40?: number;
  viscosity_cSt_100?: number;
  iron_ppm?: number; copper_ppm?: number; lead_ppm?: number;
  aluminum_ppm?: number; silicon_ppm?: number; sodium_ppm?: number;
  soot_pct?: number; water_pct?: number; tbn?: number; tan?: number;
  notes?: string[];
}

function grab(re: RegExp, s: string): number | undefined {
  const m = s.match(re); if (!m) return;
  const v = parseFloat(m[1].replace(',', '.'));
  return isNaN(v) ? undefined : v;
}

export function parseOilReport(raw: string): OilReport {
  const s = raw.replace(/\s+/g, ' ');
  const out: OilReport = {};
  out.viscosity_cSt_40  = grab(/viscosity[^0-9]{0,20}40[^0-9]{0,5}([0-9]+(?:\.[0-9]+)?)/i, s);
  out.viscosity_cSt_100 = grab(/viscosity[^0-9]{0,20}100[^0-9]{0,5}([0-9]+(?:\.[0-9]+)?)/i, s);
  out.iron_ppm    = grab(/iron[^0-9]{0,10}([0-9]+(?:\.[0-9]+)?)/i, s);
  out.copper_ppm  = grab(/copper[^0-9]{0,10}([0-9]+(?:\.[0-9]+)?)/i, s);
  out.lead_ppm    = grab(/lead[^0-9]{0,10}([0-9]+(?:\.[0-9]+)?)/i, s);
  out.aluminum_ppm= grab(/alum(?:inium)?[^0-9]{0,10}([0-9]+(?:\.[0-9]+)?)/i, s);
  out.silicon_ppm = grab(/silicon[^0-9]{0,10}([0-9]+(?:\.[0-9]+)?)/i, s);
  out.sodium_ppm  = grab(/sodium[^0-9]{0,10}([0-9]+(?:\.[0-9]+)?)/i, s);
  out.soot_pct    = grab(/soot[^0-9]{0,10}([0-9]+(?:\.[0-9]+)?)\s*%/i, s);
  out.water_pct   = grab(/water[^0-9]{0,10}([0-9]+(?:\.[0-9]+)?)\s*%/i, s);
  out.tbn         = grab(/\bTBN\b[^0-9]{0,10}([0-9]+(?:\.[0-9]+)?)/i, s);
  out.tan         = grab(/\bTAN\b[^0-9]{0,10}([0-9]+(?:\.[0-9]+)?)/i, s);
  out.notes = [];
  if (out.iron_ppm && out.iron_ppm > 60) out.notes!.push("High wear metal: iron");
  if (out.copper_ppm && out.copper_ppm > 30) out.notes!.push("Elevated copper (bearings)");
  if (out.soot_pct && out.soot_pct > 2.5) out.notes!.push("High soot %");
  if (out.water_pct && out.water_pct > 0.3) out.notes!.push("Water contamination risk");

  return out;
}

6) API Routes



Add server/routes/rag.ts and mount under /api:

import { Router } from "express";
import multer from "multer";
import { ingestFileToKb, searchKb } from "../rag/ingest";
import { embedBatch } from "../rag/embeddings";
import { parseOilReport } from "../rag/oil-report";

const upload = multer({ dest: "uploads/" });
export const ragRouter = Router();

// Upload & ingest
ragRouter.post("/kb/ingest", upload.single("file"), async (req: any, res) => {
  try {
    if (!req.file) return res.status(400).json({ error: "file missing" });
    const orgId = req.orgId || null;
    const result = await ingestFileToKb(req.file.path, orgId);
    res.json({ success: true, ...result });
  } catch (e) {
    console.error(e);
    res.status(500).json({ error: "ingestion failed" });
  }
});

// Semantic search
ragRouter.post("/kb/search", async (req: any, res) => {
  const { query, k = 5 } = req.body || {};
  if (!query) return res.status(400).json({ error: "query is required" });
  const [qemb] = await embedBatch([query]);
  const rows: any = await searchKb(req.orgId || null, qemb, k);
  res.json({ results: rows?.rows || [] });
});

// Quick diagnosis: retrieve top K chunks and echo
ragRouter.post("/kb/diagnose", async (req: any, res) => {
  const { symptom, k = 5 } = req.body || {};
  if (!symptom) return res.status(400).json({ error: "symptom is required" });
  const [qemb] = await embedBatch([symptom]);
  const rows: any = await searchKb(req.orgId || null, qemb, k);
  res.json({
    inputs: { symptom },
    evidence: rows?.rows || [],
    // Keep a trivial heuristic summary for now
    summary: "Top related manual/passages returned. Use LLM to synthesize actions if available."
  });
});

// Parse OCR Oil Report text (client can send raw extracted text too)
ragRouter.post("/kb/oil-report/parse", async (req: any, res) => {
  const { raw } = req.body || {};
  if (!raw) return res.status(400).json({ error: "raw text required" });
  const parsed = parseOilReport(raw);
  res.json({ parsed });
});

7) Minimal Frontend: “Knowledge” Page (Uploader + Search)



Create client/src/pages/knowledge.tsx:

import { useState } from "react";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";

export default function Knowledge() {
  const [file, setFile] = useState<File|null>(null);
  const [query, setQuery] = useState("");
  const [results, setResults] = useState<any[]>([]);

  const ingest = async () => {
    if (!file) return;
    const form = new FormData();
    form.append("file", file);
    const r = await fetch("/api/kb/ingest", { method: "POST", body: form });
    const j = await r.json();
    alert(j.success ? `Ingested: ${j.chunks} chunks` : `Error: ${j.error}`);
  };

  const search = async () => {
    const r = await fetch("/api/kb/search", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ query, k: 5 })
    });
    const j = await r.json();
    setResults(j.results || []);
  };

  return (
    <div className="container mx-auto p-6 space-y-6">
      <Card>
        <CardHeader><CardTitle>Knowledge Base</CardTitle></CardHeader>
        <CardContent className="space-y-4">
          <div className="flex gap-2">
            <Input type="file" onChange={(e)=> setFile(e.target.files?.[0] || null)} />
            <Button onClick={ingest}>Upload & Ingest</Button>
          </div>
          <div className="flex gap-2">
            <Input placeholder="Search manuals or oil reports…" value={query} onChange={e=>setQuery(e.target.value)} />
            <Button variant="outline" onClick={search}>Search</Button>
          </div>
          <div className="space-y-2">
            {results.map((r, i)=>(
              <div key={i} className="p-3 rounded border">
                <div className="text-xs text-muted-foreground">{r.name} · #{r.ord}</div>
                <div className="mt-1 text-sm whitespace-pre-wrap">{r.text}</div>
              </div>
            ))}
          </div>
        </CardContent>
      </Card>
    </div>
  );
}
