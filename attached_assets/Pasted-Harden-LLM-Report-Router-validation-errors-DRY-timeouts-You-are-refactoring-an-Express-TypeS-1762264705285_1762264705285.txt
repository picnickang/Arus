import type { Response } from "express";

export function sendOk<T>(res: Response, data: T, status = 200) {
  return res.status(status).json({ success: true, ...data });
}
export function sendErr(res: Response, message: string, status = 400, extra?: Record<string, unknown>) {
  return res.status(status).json({ success: false, error: message, ...(extra ?? {}) });
}
export const asyncHandler = <T extends (...a:any[])=>Promise<any>>(fn:T) =>
  (req:any,res:any,next:any)=>fn(req,res,next).catch(next);

// util: 30s timeout guard for long LLM operations
function withTimeout<T>(p: Promise<T>, ms = 30_000): Promise<T> {
  const ac = new AbortController();
  const timer = setTimeout(() => ac.abort(), ms);
  // If enhancedLLM supports signal, pass ac.signal; else just race:
  const timeout = new Promise<never>((_, rej) => setTimeout(() => rej(new Error(`Timed out after ${ms}ms`)), ms));
  return Promise.race([p, timeout]).finally(() => clearTimeout(timer));
}

      lastOccurrence: new Date(p.lastObserved).toISOString().split("T")[0],

    const anomalies = learnings.failurePatterns
      .filter((p) => p.patternType === "failure" && p.confidence > 0.6)
      .map((p) => ({
        type: p.patternType,
        severity: p.confidence > 0.8 ? ("critical" as const) : p.confidence > 0.6 ? ("high" as const) : ("medium" as const),
        detectedAt: new Date(p.lastObserved).toISOString(),
        description: p.description,
        zScore: p.confidence * 3,
      }));

    const seasonalTrends = learnings.maintenancePatterns
      .filter((p) => p.patternType === "seasonal" || p.patternType === "operational")
      .map((p) => ({
        season: p.patternType === "seasonal" ? "Seasonal" : "Operational",
        trend: p.description.includes("increasing") ? "increasing" : p.description.includes("decreasing") ? "decreasing" : "stable",
        impact: p.confidence > 0.7 ? "High impact on maintenance" : "Moderate impact",
      }));

    const equipmentCorrelations = learnings.failurePatterns
      .filter((p) => Array.isArray(p.affectedEquipment) && p.affectedEquipment.length >= 2)
      .slice(0, 5)
      .map((p) => ({
        equipment1: p.affectedEquipment[0] ?? "Unknown",
        equipment2: p.affectedEquipment[1] ?? "Unknown",
        correlationType: p.patternType,
        strength: p.confidence,
      }));

    const failureRisk = Math.round(
      (learnings.failurePatterns.reduce((sum, p) => sum + p.confidence, 0) /
        Math.max(learnings.failurePatterns.length, 1)) *
        100
    );

    const criticalEquipment = learnings.failurePatterns
      .filter((p) => p.confidence > 0.7)
      .flatMap((p) => p.affectedEquipment)
      .filter((v, i, a) => v && a.indexOf(v) === i)
      .slice(0, 5);

      learnings.maintenancePatterns.length > 0

        patterns: { historicalPatterns, anomalies, seasonalTrends, equipmentCorrelations },

3) Add retry/backoff helper

private async withBackoff<T>(fn: () => Promise<T>, retries = 2, base = 300): Promise<T> {
  let attempt = 0;
  for (;;) {
    try {
      return await fn();
    } catch (e: any) {
      const status = e?.status ?? e?.code ?? 0;
      const retriable = [429, 500, 502, 503, 504].includes(Number(status));
      if (attempt >= retries || !retriable) throw e;
      const delay = base * Math.pow(2, attempt) + Math.random() * 100;
      await new Promise(r => setTimeout(r, delay));
      attempt++;
    }
  }
}

private serializeContext(context: ReportContext): string {

  const parts: string[] = [];

      urgent: context.data.workOrders.filter((wo: any) => wo.priority === 'urgent').length,
      normal: context.data.workOrders.filter((wo: any) => wo.priority === 'normal').length
    };
    parts.push(`  - Critical: ${byPriority.critical}, Urgent: ${byPriority.urgent}, Normal: ${byPriority.normal}`);
  }

  if (context.intelligence?.knowledgeBase?.length) {
    parts.push('\nKey Insights:');
    for (const snippet of context.intelligence.knowledgeBase.slice(0, 5)) {
      parts.push(`  - ${snippet}`);
    }
  }

  if (parts.length > MAX_LINES) {
    parts.length = MAX_LINES - 1;
