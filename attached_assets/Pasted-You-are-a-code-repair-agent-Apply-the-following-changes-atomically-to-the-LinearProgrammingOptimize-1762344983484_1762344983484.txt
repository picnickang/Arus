// ===== PRECEDENCE CONSTRAINTS (A must finish before B starts) =====
for (let succIdx = 0; succIdx < jobs.length; succIdx++) {
  const succJob = jobs[succIdx];
  if (!succJob.dependencies || succJob.dependencies.length === 0) continue;

  const succVars = varIndexByJob.get(succIdx) || [];
  for (const depId of (succJob.dependencies || [])) {
    const predIdx = jobs.findIndex(j => j.id === depId);
    if (predIdx < 0) continue; // dep not in this batch

    const predVars = varIndexByJob.get(predIdx) || [];
    if (predVars.length === 0 || succVars.length === 0) continue;

    // For each pair (predVar, succVar) that temporally conflicts (pred end > succ start),
    // add: predVar + succVar <= 1  (so succ can't start before pred finishes)
    for (const sVar of succVars) {
      for (const pVar of predVars) {
        // Same day: pred must end at/before succ start
        // Cross-day: any pred day > succ day obviously ends after succ start
        const predEndsAfterSuccStarts =
          (pVar.day > sVar.day) ||
          (pVar.day === sVar.day && pVar.endHour > sVar.startHour);

        if (!predEndsAfterSuccStarts) continue;

        // Unique constraint name per conflicting pair
        const cname = `prec_j${predIdx}_to_j${succIdx}_pd${pVar.day}ph${pVar.startHour}_sd${sVar.day}sh${sVar.startHour}`;
        // define max 1
        if (!constraintDefs[cname]) constraintDefs[cname] = { max: 1 };

        // set coeff 1 on both variables in this constraint
        // (variables already exist; just augment their coefficient rows)
        variables[pVar.name][cname] = 1;
        variables[sVar.name][cname] = 1;
      }
    }
  }
