import cron from 'node-cron';
import { storage } from './storage';

// Per-process guard (does NOT protect across replicas)
let vesselSchedulerInitialized = false;

const TZ = 'Asia/Singapore';
const DEFAULT_CRON = '0 0 * * *'; // midnight daily (SGT)

// ---- Time helpers ----
function toTzDate(d = new Date(), tz = TZ): Date {
  // Convert "now" to target TZ using toLocaleString round-trip
  return new Date(new Date(d).toLocaleString('en-US', { timeZone: tz }));
}
function yyyymmdd(d: Date): string {
  const yyyy = d.getFullYear();
  const mm = String(d.getMonth() + 1).padStart(2, '0');
  const dd = String(d.getDate()).padStart(2, '0');
  return `${yyyy}-${mm}-${dd}`;
}
function startOfDayTZ(d = new Date(), tz = TZ): Date {
  const z = toTzDate(d, tz);
  return new Date(z.getFullYear(), z.getMonth(), z.getDate());
}
function endOfDayTZ(d = new Date(), tz = TZ): Date {
  const s = startOfDayTZ(d, tz);
  return new Date(s.getTime() + 24 * 60 * 60 * 1000 - 1);
}

/**
 * Business rule:
 * - Count **downtimeDays** if the vessel had any downtime in the analysis window for "yesterday" (SGT),
 *   based on work orders that affect vessel downtime and report any downtime hours (> 0).
 * - Otherwise count **operationDays**.
 * - Never increment both on the same day.
 * - Idempotent per vessel per date (SGT) via `lastDailyUpdateDate`.
 */
export function setupVesselSchedules(): void {
  if (vesselSchedulerInitialized) {
    console.log('⚠️ Vessel operation scheduler already initialized, skipping...');
    return;
  }

  const vesselOperationSchedule = process.env.VESSEL_OPERATION_CRON || DEFAULT_CRON;
  if (!cron.validate(vesselOperationSchedule)) {
    console.error(`❌ Invalid VESSEL_OPERATION_CRON "${vesselOperationSchedule}". Falling back to "${DEFAULT_CRON}".`);
  }

  console.log('⚓ Setting up vessel operation+downtime schedules...');

  const task = cron.schedule(
    cron.validate(vesselOperationSchedule) ? vesselOperationSchedule : DEFAULT_CRON,
    async () => {
      const nowTz = toTzDate(new Date(), TZ);

      // Analyze **yesterday** in SGT
      const yStart = startOfDayTZ(new Date(nowTz.getTime() - 24 * 60 * 60 * 1000), TZ);
      const yEnd = endOfDayTZ(yStart, TZ);
      const dayKey = yyyymmdd(yStart); // YYYY-MM-DD in SGT

      const lockKey = `locks:vessel-daily-counters:${dayKey}`;

      const gotLock = await safeAcquireLock(lockKey, 60 * 60); // 1h TTL
      if (!gotLock) {
        console.log(`🔒 Another worker holds lock ${lockKey}. Skipping run.`);
        return;
      }

      console.log(`⚓ Daily counters starting for ${dayKey} (${TZ})...`);

      try {
        const [allVessels, allWorkOrders] = await Promise.all([
          storage.getVessels(),
          storage.getWorkOrders(), // if available: storage.getWorkOrders({ since: yStart, until: yEnd })
        ]);

        const active = allVessels.filter(v => v.active);
        console.log(`Found ${active.length} active vessels out of ${allVessels.length} total`);

        // Build a map of downtime presence for each vessel for yesterday
        const isDowntimeDay: Record<string, boolean> = {};
        for (const wo of allWorkOrders) {
          // Filter to yesterday window
          const created = new Date(wo.createdAt);
          if (created < yStart || created > yEnd) continue;

          const affects = !!wo.affectsVesselDowntime;
          const hours = Number(wo.actualDowntimeHours ?? wo.estimatedDowntimeHours ?? 0);
          if (!affects || hours <= 0) continue;

          if (wo.vesselId) isDowntimeDay[wo.vesselId] = true;
        }

        let incOps = 0;
        let incDown = 0;
        let skippedIdempotent = 0;

        const updates = active.map(async (v) => {
          try {
            // Idempotency per day (SGT)
            if (v.lastDailyUpdateDate === dayKey) {
              skippedIdempotent++;
              return;
            }

            // Decide which counter to increment
            const downtime = !!isDowntimeDay[v.id];

            // Prefer atomic increment if storage supports it
            if (typeof (storage as any).incrementVesselField === 'function') {
              const field = downtime ? 'downtimeDays' : 'operationDays';
              await (storage as any).incrementVesselField(v.id, field, 1, {
                extra: { lastDailyUpdateDate: dayKey },
              });
            } else {
              const op = Number(v.operationDays ?? 0);
              const down = Number(v.downtimeDays ?? 0);
              await storage.updateVessel(v.id, {
                operationDays: downtime ? op : op + 1,
                downtimeDays: downtime ? down + 1 : down,
                lastDailyUpdateDate: dayKey,
              });
            }

            if (downtime) incDown++; else incOps++;
          } catch (err) {
            console.error(`❌ Failed to update daily counters for vessel ${v.id}:`, err);
          }
        });

        await Promise.allSettled(updates);

        console.log(
          `✅ Daily counters (${dayKey}) done: operationDays+=${incOps}, downtimeDays+=${incDown}, skipped=${skippedIdempotent}, active=${active.length}`
        );
      } catch (error) {
        console.error('❌ Vessel daily counters job failed:', error);
      } finally {
        await safeReleaseLock(lockKey);
      }
    },
    { timezone: TZ }
  );

  // Graceful shutdown
  process.once('SIGINT', () => task.stop());
  process.once('SIGTERM', () => task.stop());

  vesselSchedulerInitialized = true;
  console.log(`✅ Vessel daily schedule configured (${vesselOperationSchedule} ${TZ})`);
}

// ---- Lock helpers (adapt to your storage/Redis) ----
async function safeAcquireLock(key: string, ttlSeconds: number): Promise<boolean> {
  try {
    if (typeof (storage as any).acquireLock === 'function') {
      return await (storage as any).acquireLock(key, ttlSeconds);
    }
    console.warn('⚠️ No distributed lock available; proceeding without lock (multi-process risk).');
    return true;
  } catch (e) {
    console.error('Lock acquisition error; proceeding without lock:', e);
    return true;
  }
}
async function safeReleaseLock(key: string): Promise<void> {
  try {
    if (typeof (storage as any).releaseLock === 'function') {
      await (storage as any).releaseLock(key);
    }
  } catch (e) {
    console.error('Lock release error:', e);
  }
}
What this adds

SG timezone execution.

Idempotency per vessel/day via lastDailyUpdateDate.

Mutual exclusivity: increments either operationDays or downtimeDays (never both).

Downtime detection from work orders in yesterday’s window:

affectsVesselDowntime === true

actualDowntimeHours or estimatedDowntimeHours > 0

Distributed lock (uses storage.acquireLock/releaseLock if available).

Numeric counters (prefers atomic incrementVesselField if supported).
