export type RawFrame = {
  id: number;
  ts: number;              // unix ms
  source: string;          // "can0", "serial1", "mqtt:engine-room"
  protocol: string;        // "J1939" | "J1587" | "SERIAL" | "MQTT" | etc

  qualityFlags?: number;
};

export type DecodeContext = {
  // Optional: map bus->equipmentId or source->equipmentId (vessel-specific mapping)
  resolveEquipmentId?: (source: string) => string | null;

  // Optional: data quality / provenance
  defaultEquipmentId?: string;

  // Optional: for consistent timestamping/clock skew handling
  nowMs?: () => number;
};

server/telemetry/decode/index.ts

import { RawFrame, DecodeContext } from "./types";
import { decodeJ1939 } from "./j1939";
import { decodeJ1587 } from "./j1587";

export function decodeFrame(frame: RawFrame, ctx: DecodeContext = {}) {
  const protocol = (frame.protocol || "").toUpperCase();

  if (protocol === "J1939") return decodeJ1939(frame, ctx);
  if (protocol === "J1587" || protocol === "J1708") return decodeJ1587(frame, ctx);

  // Unknown protocol: return empty so pipeline remains stable.
  return [];
}

2) J1939 decoder skeleton (PGN parsing + SPN extraction)

This assumes you store the CAN header somewhere. You have two workable options:

Option A (recommended): agent payload format is a compact envelope

Have the agent write payload as:
	•	uint32 canId (little-endian) + uint8 dlc + dlc bytes data
That makes PGN extraction deterministic.

server/telemetry/decode/j1939.ts

import { RawFrame, DecodeContext } from "./types";
import { TelemetryReading } from "../telemetry-batch-writer"; // adjust import path if needed
import { readU32LE, clamp } from "./util";
import { decodePgn } from "./registry";

// Payload format: [canId:u32le][dlc:u8][data:dlc bytes]
export function decodeJ1939(frame: RawFrame, ctx: DecodeContext): TelemetryReading[] {
  const buf = frame.payload;
  if (buf.length < 6) return [];

  const canId = readU32LE(buf, 0);
  const dlc = buf.readUInt8(4);
  const data = buf.subarray(5, 5 + clamp(dlc, 0, 8));
  if (data.length === 0) return [];

  // J1939 29-bit ID fields
  // Priority: bits 26-28
  // PGN: bits 8-25 (but PDU1 vs PDU2 rules)
  const pgn = extractPgn(canId);
  const sa = canId & 0xff; // source address

  const equipmentId =
    ctx.resolveEquipmentId?.(frame.source) ??
    ctx.defaultEquipmentId ??
    "unknown";

  // Decode via registry (PGN -> readings)
  return decodePgn({
    pgn,
    sa,
    data,
    ts: frame.ts,
    equipmentId,
    source: frame.source,
  });
}

function extractPgn(canId: number): number {
  const pf = (canId >> 16) & 0xff;
  const ps = (canId >> 8) & 0xff;

  // PDU1 (pf < 240): PGN includes PS = 0
  // PDU2 (pf >= 240): PGN includes PS
  const pgn = pf < 240 ? (pf << 8) : (pf << 8) | ps;
  return pgn;
}

3) Registry-based PGN/SPN decoding (incremental rollout)

server/telemetry/decode/registry.ts

import { TelemetryReading } from "../telemetry-batch-writer";

type PgnDecodeInput = {
  pgn: number;
  sa: number;
  data: Buffer;
  ts: number;
  equipmentId: string;
  source: string;
};

type PgnDecoder = (i: PgnDecodeInput) => TelemetryReading[];

const registry = new Map<number, PgnDecoder>();

export function registerPgn(pgn: number, fn: PgnDecoder) {
  registry.set(pgn, fn);
}

export function decodePgn(i: PgnDecodeInput): TelemetryReading[] {
  const fn = registry.get(i.pgn);
  if (!fn) return [];
  try {
    return fn(i);
  } catch {
    return [];
  }
}

/**
 * Example PGN decoder: Engine Speed (commonly SPN 190, often PGN 61444 / 0x00F004 in many datasets)
 * NOTE: exact PGN/SPN mapping depends on your engine + DBC/J1939 database.
 */
registerPgn(0x00F004, (i) => {
  // This is intentionally “template-like”; replace with your real DBC/SPN definitions.
  // Example: interpret bytes [3..4] little-endian as rpm*0.125 (this is a common J1939 scaling for SPN190)
  if (i.data.length < 5) return [];
  const raw = i.data.readUInt16LE(3);
  const rpm = raw * 0.125;

  return [{
    equipmentId: i.equipmentId,
    sensorType: "ENGINE_SPEED_RPM",
    value: rpm,
    timestamp: new Date(i.ts),
    source: i.source,
    // add other fields your TelemetryReading requires
  } as any];
});

This lets you add PGNs one-by-one without touching the bridge again.

export function readU32LE(buf: Buffer, offset: number) {
  if (offset + 4 > buf.length) return 0;
  return buf.readUInt32LE(offset);
}
export function clamp(n: number, min: number, max: number) {
  return Math.max(min, Math.min(max, n));

import Database from "better-sqlite3";
import { TelemetryBatchWriter } from "../../telemetry-batch-writer";
import { pool } from "../../db-config";
import { decodeFrame } from "../../telemetry/decode";
import type { RawFrame } from "../../telemetry/decode/types";

const SQLITE_BATCH = 2000;

export async function runSqliteBridge(sqlitePath: string) {
  const db = new Database(sqlitePath);
  db.pragma("journal_mode = WAL");
  db.pragma("busy_timeout = 5000");

  db.exec(`
    CREATE TABLE IF NOT EXISTS ingest_cursor (
      key TEXT PRIMARY KEY,
      last_raw_frame_id INTEGER NOT NULL
    );
    INSERT OR IGNORE INTO ingest_cursor(key, last_raw_frame_id) VALUES ('raw_frames', 0);
  `);

  const getCursor = db.prepare(`SELECT last_raw_frame_id AS lastId FROM ingest_cursor WHERE key='raw_frames'`);
  const setCursor = db.prepare(`UPDATE ingest_cursor SET last_raw_frame_id=? WHERE key='raw_frames'`);

  const fetchBatch = db.prepare(`
    SELECT id, unix_time_ms, source, protocol, payload, quality_flags
    FROM raw_frames
    WHERE id > ?
    ORDER BY id
    LIMIT ?
  `);

  const writer = new TelemetryBatchWriter({ pool });

  while (true) {
    const lastId = (getCursor.get() as any)?.lastId ?? 0;
    const rows = fetchBatch.all(lastId, SQLITE_BATCH) as any[];

    if (!rows.length) {
      await sleep(500);
      continue;
    }

    const readings = [];
    for (const r of rows) {
      const frame: RawFrame = {
        id: r.id,
        ts: Number(r.unix_time_ms),
        source: r.source,
        protocol: r.protocol,
        payload: Buffer.from(r.payload),
        qualityFlags: Number(r.quality_flags ?? 0),
      };
      readings.push(...decodeFrame(frame, {
        resolveEquipmentId: (src) => mapSourceToEquipmentId(src),
        defaultEquipmentId: "unknown",
      }));
    }

    try {
      // Use your existing batching/backpressure path:
      // If TelemetryBatchWriter is queue-based, call queue() repeatedly then flush.
      // If it supports queueMany(), use it.
      for (const rd of readings) writer.queue(rd);

      await writer.flush(); // (or writer.drain(), writer.flushNow(), etc. depending on your implementation)

      const maxId = rows[rows.length - 1].id;
      setCursor.run(maxId);
    } catch (e) {
      // PG offline or transient error: do not advance cursor
      await sleep(2000);
    }
  }
}

If it’s queue() + internal timer flush (no explicit flush()), then replace that block with:
